\chapter{扩展立方体的额外连通度和额外诊断度}
%\chapter{扩展立方体的$2$-额外连通度和$MM^*$模型下的$2$-额外诊断度}

在互联网络的容错性研究中，额外连通度是衡量网络稳定性和容错性的一个重要指标，
额外连通度从更实际的角度考察了网络中可能发生故障的情形，对实际网络的性能评估提供了重要的依据。在本章中，我们重点考察了扩展立方体的$2$-额外连通度和$MM^*$模型下的$2$-额外诊断度，以及相应的诊断算法，其可以帮助我们快速的定位故障顶点来保障网络的正常运行。

本章针对扩展立方体所得到的相关结论如下：

（1）扩展立方体的$2$-额外连通度是$6n-17$，其中$n \geq 6$；

（2）扩展立方体在$MM^*$模型下的$2$-额外诊断度是$6n-5$，其中$n \geq 6$。

\section{预备知识}

在给出扩展立方体的$2$-额外连通度和$MM^*$模型下的$2$-额外诊断度的证明和算法之前，我们先给出证明要用到的扩展立方体的一些性质。


一对$AQ_n$中的顶点$(u,v)$被称为$AQ_n$中的一个补对如果$u=\bar{u}_n$。因为$AQ_n$中的任意两个补对都是两两不相交的，所以$AQ_n$中包含$2^{n-1}$个补对。

\begin{lem}
\cite{The-super-connectivity-of-augmented-cubes}
\label{complement-pair}
设$AQ_n=L \bigodot R$，其中$L \cong AQ^0_{n-1},R \cong AQ^1_{n-1}$。如果$u$和$v$是$L$ 中的$2$个顶点且$(u,v)($即，$v=\bar{u}_{n-1})$是$L$中的一个补对，那么$(u,v)$在$R$中有两个相同的邻居顶点。否则，如果$(u,v)$不是$L$中的一个补对，则$(u,v)$在$R$中没有相同的邻居顶点。
\end{lem}


\begin{lem}
\cite{Conditional-Diagnosability-of-Augmented-Cubes}
\label{5-vertices-2-complement-pairs}
当$n \geq 4$时，是$x,y,s,t,u$是$AQ_n$中的5个顶点，如果$AQ_n$在$x,y,s,t,u$包含两个补对，$|N_{AQ_n}(\{x,y,s,t,u\})| \geq 10n-35$。
\end{lem}

\begin{lem}
\cite{Conditional-Diagnosability-of-Augmented-Cubes}
\label{5-vertices-1-complement-pairs}
当$n \geq 4$时，设$x,y,s,t,u$是$AQ_n$中的5个顶点，如果$AQ_n$在$x,y,s,t,u$包含一个补对，$|N_{AQ_n}(\{x,y,s,t,u\})| \geq 10n-39$。
\end{lem}

\begin{lem}
\cite{Conditional-Diagnosability-of-Augmented-Cubes}
\label{6-vertices-3-complement-pairs}
当$n \geq 4$时，设$x,y,s,t,u,v$是$AQ_n$中的6个顶点，如果$AQ_n$在$x,y,s,t,u,v$包含三个补对，$|N_{AQ_n}(\{x,y,s,t,u,v\})| \geq 12n-46$。
\end{lem}

\begin{lem}
\cite{t/t-Diagnosability-and-t/k-Diagnosability}
\label{k-nodes}
对于顶点集合$X \subset V(AQ_n)$，其中$|X|=k, 0 \leq k \leq 2n-1$，那么$|N_{AQ_n}(X)| \geq 2kn- \frac{3k(k+1)}{2}+1$。
\end{lem}


\begin{thm}
\cite{Conditional-Diagnosability-of-Augmented-Cubes}
\label{4n-9<=}
设$S$是$AQ_n$的一个顶点子集，其中$|S| \leq 4n-9$, $n \geq 4$。那么, $AQ_n-S$满足以下两个条件之一:

(1) $AQ_n-S$是连通的；

(2) $AQ_n-S$有两个分支，其中一个是$K_1$，另一个包含$2^n-|S|-1$个顶点。
\end{thm}

\begin{thm}
\cite{Conditional-Diagnosability-of-Augmented-Cubes}
\label{6n-18<=}
设$S$是$AQ_n$的一个顶点子集，其中$|S| \leq 6n-18$, $n \geq 6$。那么, $AQ_n-S$满足以下四个条件之一:

(1) $AQ_n-S$是连通的；

(2) $AQ_n-S$有两个分支，其中一个是$K_1$，另一个包含$2^n-|S|-1$个顶点；

(3) $AQ_n-S$有两个分支，其中一个是$K_2$，另一个包含$2^n-|S|-2$个顶点；

(4) $AQ_n-S$有三个分支，其中有两个是$K_1$，且第三个包含$2^n-|S|-2$个顶点。
\end{thm}



\section{扩展立方体的$2$-额外连通度}


\begin{defn}
\label{extra diagnosability definition}
\cite{FABREGA199649}
(1) 图$G$的一个$g$-额外集是一个顶点的集合，当从$G$中删除这个集合后，每个连通分量所包含的顶点个数大于等于$g+1$。

(2) 图$G$的一个$g$-额外割是$G$的一个$g$-额外集，将其删除将会导致$G$变为不连通的。

(3) 图$G$的$g$-额外连通度，记为$\kappa_g(G)$, 是$g$-额外割的最小基数。
\end{defn}

因为条件诊断度和$2$-额外连通度具有密切的联系，在得到条件诊断度或者$2$-额外连通度的条件下，如果满足某些条件，另一个诊断度可以直接得到。Hao\cite{Relationship-between-Conditional-and-extra}等人给出了的充分条件。

我们首先给出条件诊断度的定义：

\begin{defn}
\cite{Conditional-Diagnosability-Measures-for-Large}
一个系统$G$是条件$t$-可诊断的，当且仅当两个不同的条件故障集合$F_1,F_2 \subseteq V(G)$ 是可区分的，且$|F_1|,|F_2| \leq t$。
\end{defn}

$G$的条件诊断度，记为$t_c(G)$，定义为$t$的最大值，使得$G$是条件$t$-诊断的。

\begin{lem}
\cite{Strong-diagnosability-and-conditional-diagnosability}
$t_c(AQ_3)=3$，$t_c(AQ_4)=7$，$t_c(AQ_5)=12$；当$n \geq 6$时，$t_c(AQ_n)=6n-17$。
\end{lem}


\begin{thm}
\label{conditional-and-extra}
\cite{Relationship-between-Conditional-and-extra}
设$G$为$n$-正则图，其中$t=$min$\{|N(T)|:T$ 是$G$中一条长度为$3$的路径或者长度为$3$ 的圈$\}$。 如果$G$满足下面的条件：

(1) 对于一个顶点集合$F \subset V(G)$，$|F|\leq t-1$，$G-F$有一个大的连通分量和小的连通分量，且这些小的连通分量的所有顶点个数最多为$2$个；

(2) $n \geq 2\gamma(G)+2$ 如果$G$中不包含长度为$5$的圈；或者$n \geq 3\gamma(G)+2$；

(3) $|V(G)|>(n+1)(t-1)+4$；

则$t_c(G)=t=\kappa_2(G)$。
\end{thm}

其中$\gamma(G)$的定义为：对于$G$中两个不同的顶点$x$和$y$，$\gamma(x,y)=|N(x) \cap N(y)|$，$\gamma(G)=$max$\{\gamma(x,y):x,y \in V(G)$ and $xy \notin E(G)\}$。

下面我们将根据扩展立方体的性质来逐一验证这$3$个条件。

$AQ_n$是$(2n-1)$-正则图。设$u$是$AQ_n$中的任意一个顶点，$T=\{u,\bar{u}_{n-1},\bar{u}_{n-3}\}$，根据扩展立方体的定义，$T$是一条长度为$3$的路径且$N(T)=6n-17$。 根据引理\ref{3nodes}，可知$AQ_n$符合条件（1）；根据扩展立方体的定义，$AQ_n$包含长度为$5$ 的圈。根据引理\ref{common_verticles}，$\gamma(AQ_n)=4$。当$n \geq 8$时，$2n-1 \geq 3\gamma(G)+2$，条件（2）成立；当$n \geq 9$时，$|V(G)|=2^n>(n+1)(t-1)+4=(n+1)(6n-18)+4$。条件（3）成立。

因为$AQ_n$满足定理\ref{conditional-and-extra}的条件，所以我们可以直接得到$AQ_n$的$2$-额外连通度。

\begin{thm}
当$n \geq 9$时，$\kappa_2(AQ_n)=6n-17$。
\end{thm}

\section{扩展立方体在$MM^*$模型下的$2$-额外诊断度}

\begin{lem}
\cite{Conditional-Diagnosability-Measures-for-Large,On-self-diagnosable-multiprocessor}
\label{MM*-distinguable}
对于图$G$中的两个不同的顶点子集$F_1 \neq F_2 \subseteq V(G)$。 $(F_1,F_2)$在$MM^*$模型下是可区分的当且仅当下面三个条件中的一个成立（见图\ref{MM*}）：

(1) $\exists x,z \in V(G)-(F_1 \cup F_2)$和$y \in (F_1-F_2) \cup (F_2-F_1)$，且$xz,yz \in E(G)$；

(2) $\exists x,y \in F_1-F_2$和$z \in V(G)-(F_1 \cup F_2)$，且$xz,yz \in E(G)$；

(3) $\exists x,y \in F_2-F_1$和$z \in V(G)-(F_1 \cup F_2)$，且$xz,yz \in E(G)$。
\end{lem}

\begin{defn}
\cite{The-g-extra-conditional-sequential}
\label{MM*-distinguishable}
(1) 在$MM^*$模型下，一个图$G$是$g$-额外，$t$-可诊断的，当且仅当对任意两个不同的$g$-额外集$F_1 \subseteq V(G)$，$F_2 \subseteq V(G)$，其中$|F_1| \leq t$，$|F_2| \leq t$且$F_1$和$F_2$是可区分的。

(2) $G$在$MM^*$下的$g$-额外诊断度记为$t^{\tilde{m}}_h(G)$，其中$t$是$G$ 在$g$-额外，$t$-可诊断下的最大值。
\end{defn}


在写这部分内容的时候，已经有两篇文章\cite{Lv-extra-connectivity-and-extra-diagnosability,A-New-Proof-for-Exact-Relationship} 在某些条件下将额外连通度和额外诊断度联系了起来。但是扩展立方体并不满足这两篇文章的条件。下面我们将对这两篇文章的条件进行逐一说明:

(1) \cite{Lv-extra-connectivity-and-extra-diagnosability}针对的是$n$-连接的网络来建立$MM^*$模型下额外连通度和额外诊断度的联系。但是扩展立方体的$2n-1$连接的网络，因此不符合该论文的适用条件。

(2) \cite{A-New-Proof-for-Exact-Relationship}中的定理1对于$MM^*$模型下将额外连通度和额外诊断度的联系起来有$3$个条件，其中扩展立方体满足其中的第$1$和第$2$个条件，但是对于第$3$个条件，并没有已知的定理满足，因此也无法将这三个条件适用于扩展立方体，以此来根据额外连通度来直接得到扩展立方体的额外诊断度。

因为没有已知的条件将$2$-额外连通度和$2$-额外诊断度联系起来，所以下面我们将自己来证明扩展立方体的$2$-额外诊断度。


\begin{lem}
\label{6-vertices-2-complement-pairs}
是$x,y,s,t,u,v$是$AQ_n(n \geq 4)$中的6个顶点，如果$AQ_n$在$x,y,s,t,u,v$包含2个补对，则$|N_{AQ_n}(\{x,y,s,t,u,v\})| \geq 12n-48$。
\end{lem}

\begin{proof}
设$(x,y),(s,t)$是$AQ_n=L \bigodot R$中的2个补对，其中$L \cong AQ^0_{n-1},R \cong AQ^1_{n-1}$。不失一般性，我们设$x,s \in V(L),y,t\in V(R)$。然后我们有下面几种情况：

\textbf{情形1}：$u,v \in L$或$u,v \in R$中。不失一般性，我们假设$u,v \in L$，根据引理\ref{4nodes}和引理\ref{2nodes}，我们有$|N_L(\{x,s,u,m\})| \geq 8(n-1)-28=8n-36,|N_L(\{y,t\})|=4(n-1)-8$。因此
\begin{align*}
|AQ_n(\{x,y,s,t,u,v\})| &\geq |N_L(\{x,s,u,v\})|+|N_R(\{y,t\})|\\
                        &\geq (8n-36)+(4n-12)=12n-48.
\end{align*}

\textbf{情形2}：$u,v$其中一个在$L$中，另一个在$R$中。
根据引理\ref{3nodes}，我们有$|N_L(\{x,s,u\})| \geq 6(n-1)-17=6n-23,|N_R(\{y,t,v\})| \geq 6(n-1)-17=6n-23$，因此
\begin{align*}
|AQ_n(\{x,y,s,t,u,v\})| &\geq |N_L(\{x,s,u\})|+|N_R(\{y,t,v\})|\\
                            &\geq 2(6n-23)=12n-46.
\end{align*}
\end{proof}



\begin{lem}
\label{7-vertices-3-complement-pairs}
是$x,y,s,t,u,v,m$是$AQ_n(n \geq 4)$中的7个顶点，如果$AQ_n$在$x,y,s,t,u,v,m$包含3个补对，则$|N_{AQ_n}(\{x,y,s,t,u,v,m\})| \geq 14n-59$。
\end{lem}


\begin{proof}
设$(x,y),(s,t),(u,v)$是$AQ_n=L \bigodot R$中的3个补对，其中$L \cong AQ^0_{n-1},R \cong AQ^1_{n-1}$。不是一般性，我们设$x,s,u \in V(L),y,t,v\in V(R)$。$m$属于$L$和$R$ 中的一个，不失一般性，我们设$m \in V(L)$。根据引理\ref{4nodes}和引理\ref{3nodes}，我们有$|N_L(\{x,s,u,m\})| \geq 8(n-1)-28=8n-36,|N_L(\{y,t,v\})|=6(n-1)-17=6n-23$。因此
\begin{align*}
|AQ_n(\{x,y,s,t,u,v,m\})| &\geq |N_L(\{x,s,u,m\})|+|N_R(\{y,t,v\})|\\
                        &\geq (8n-36)+(6n-23)=14n-59.
\end{align*}

\end{proof}

\begin{lem}
\label{8-vertices-4-complement-pairs}
是$x,y,s,t,u,v,m,n$是$AQ_n(n \geq 4)$中的8个顶点，如果$AQ_n$在$x,y,s,t,u,v,m,n$包含4个补对，则$|N_{AQ_n}(\{x,y,s,t,u,v,m,n\})| \geq 16n-72$。
\end{lem}

\begin{proof}
设$(x,y),(s,t),(u,v),(m,n)$是$AQ_n=L \bigodot R$中的4个补对，其中$L \cong AQ^0_{n-1},R \cong AQ^1_{n-1}$。不是一般性，我们设$x,s,u,m \in V(L),y,t,v,n \in V(R)$。根据引理\ref{4nodes}，我们有$|N_L(\{x,s,u,m\})| \geq 8(n-1)-28=8n-36,|N_R(\{y,t,v,n\})| \geq 8(n-1)-28=8n-36$，因此
\begin{align*}
|AQ_n(\{x,y,s,t,u,v,m,n\})| &\geq |N_L(\{x,s,u,m\})|+|N_R(\{y,t,v,n\})|\\
                            &\geq 2(8n-36)=16n-72.
\end{align*}
\end{proof}


\begin{lem}
\label{2-extra diagnosability <=}
对于$n \geq 6$，$t^{\tilde{m}}_2(AQ_n) \leq 6n-15$。
\end{lem}

\begin{proof}
设$S=\{u,\bar{u}_{n-1},\bar{u}_{n-3}\}$，$F_1=N(V(S))$和$F_2=N[(V(S))]$。根据定义，我们有$|F_1|=6n-17$和$|F_2|=N(V(S))+|V(S)|=6n-14$。$AQ_n[S]$是一个连通子图且$AQ_n-F_1$中的任意一个连通分量的顶点数都大于等于3。显然，我们有

$|F_1| \leq 6n-14$, $|F_2| \leq 6n-14$。\\
并且$F_1$和$F_2$都是$AQ_n$的$2$-额外集（见图\ref{F_1andF_2-extra-sets}）。

因为$F_1 \Delta F_2=V(S)$，$E[F_1 \Delta F_2,V(AQ_n)-(F_1 \cup F_2)]=\emptyset$。根据引理\ref{MM*}，$(F_1,F_2)$在\\
$MM^*$下是不可区分的。由定义\ref{MM*-distinguishable}，$AQ_n$不是$MM^*$下$2$-额外，$(6n-14)$-可诊断的，因此$t^{\tilde{m}}_2(AQ_n) \leq 6n-15$。
\end{proof}

接下来，我们证明$t^{\tilde{m}}_2(AQ_n) \geq 6n-15$。

\begin{lem}
\label{2-extra diagnosability >=}
对于$n \geq 8$，$t^{\tilde{m}}_2(AQ_n) \geq 6n-15$。
\end{lem}

\begin{proof}
我们将通过反证法来证明$t^{\tilde{m}}_2(AQ_n) \geq 6n-15$。假设$t^{\tilde{m}}_2(AQ_n) \leq 6n-16$。设$F_1$和$F_2$是两个不同的$2$-额外集且$|F_1|=|F_2| = 6n-15$。则在$MM^*$模型下，$F_1$ 和$F_2$ 是不可区分的。另外，对于$n \geq 8$
\begin{align*}
|V(AQ_n)|-|F_1 \cup F_2| &\geq |V(AQ_n)|-|F_1|-|F_2|\\
                          &\geq |V(AQ_n)|-2(6n-15)\\
                          &\geq 2^n-2(6n-15)\\
                          &> 0
\end{align*}
因此，$V(AQ_n)-(F_1 \cup F_2) \neq \emptyset$。

\textbf{断言：}$AQ_n-(F_1 \cup F_2)$不包含孤立顶点。

假设在$AQ_n-(F_1 \cup F_2)$存在一个孤立顶点$w$。设$W$是$AQ_n-(F_1 \cup F_2)$中孤立顶点的集合，并且$H$是$AQ_n-(F_1 \cup F_2)$的导出子图。因为$F_1$是$AQ_n$的一个$2$-额外集且$(F_1,F_2)$在$MM^*$模型下是不可区分的。则在$F_2 - F_1$存在唯一一个顶点$u$且$uw \in E(AQ_n)$。 类似的，因为$F_2$是$AQ_n$的一个$2$-额外集且$(F_1,F_2)$ 在$MM^*$模型下是不可区分的。则在$F_1 - F_2$存在唯一一个顶点$v$且$vw \in E(AQ_n)$。 因为$AQ_n$是$2n-1$连通图，则$w$剩下的$2n-3$个邻居顶点都在$F_1 \cap F_2$ 中。此外，因为还有其他边把$F_1 \cap F_2$ 中的顶点与$W$外的其他顶点连接起来。因此我们有下面的不等式：
\begin{align*}
(2n-3)|W| &\leq \sum_{s\in F_1 \cap F_2}|N(s)|\\
          &= |F_1 \cap F_2| \times (2n-1)\\
          &\leq (6n-16) \times (2n-1)
\end{align*}
则
\begin{align*}
|W| &\leq \frac{(6n-16) \times (2n-1)}{2n-3}\\
   % &= \frac{(6n-16) \times (2n-3+2)}{2n-3}\\
    &= \frac{(6n-16) \times (2n-3)+(6n-16) \times 2}{2n-3}\\
    &= 6n-16+ \frac{6(2n-3)-14}{2n-3}\\
   % &= 6n-10-\frac{14}{2n-3}\\
    &\leq 6n-10
\end{align*}
假设$V(H)=\emptyset$，我们有
\begin{align*}
|V(AQ_n)| &= |F_1 \cup F_2|+ |W|\\
       &= |F_1| + |F_2| - |F_1 \cap F_2| + |W|\\
       &\leq 2 \times (6n-15) - (2n-3) + 6n-10\\
       &= 16n-37
\end{align*}
当$n \geq 8$时，$2^n > 16n-37$。因此，$V(H)\neq \emptyset$。

因为$W$是$AQ_n-(F_1 \cup F_2)$中所有孤立顶点的集合，并且$H$ 是$V(AQ_n)-(F_1 \cup F_2)$的导出子图，我们有$N(V(H))\cap W=\emptyset$。

假设$N(V(H)) \cap (F_1 \Delta F_2)\neq \emptyset$, 那么在$V(H)$中存在一个顶点$z$使得$z$在$F_1 \Delta F_2$中至少有1 个邻居顶点$y$。因为在$V(H)$中的任意一个顶点在$AQ_n-(F_1 \cup F_2)]$都不是一个孤立的顶点, 所以$z$在$V(H)$中也存在一个邻居顶点$x$。 因此, $x,z \in V(G)-(F_1 \cup F_2)\neq \emptyset$，$y \in F_1 \Delta F_2$ 且$xz,yz \in E(AQ_n)$。根据引理\ref{MM*-distinguable}, $F_1$ 和$F_2$是可区分的，这与$(F_1,F_2)$在$MM^*$模型下是不可区分的相矛盾。因此$N(V(H)) \cap (F_1 \Delta F_2)= \emptyset$。

如果$F_1 \cap F_2 = \emptyset$，那么$F_1 \Delta F_2=F_1 \cup F_2$。所以，$N(V(H)) \cap F_1 = \emptyset$ 和$N(V(H)) \cap F_2 = \emptyset$。进一步的，$N(V(H)) \cap (F_1 \cup F_2 \cup W) = \emptyset$。因为，$V(AQ_n)=N(V(H)) \cup F_1 \cup F_2 \cup W$，这意味着$AQ_n$ 是不连通的，这与$AQ_n$是连通图相矛盾。因此，$F_1 \cap F_2 \neq \emptyset$。

因为$F_1 \neq F_2$，不失一般性，$F_1 - F_2 \neq \emptyset$。 因为$F_2$是$AQ_n$的一个额外集，因此存在一个顶点$u \in F_1 - F_2$使得$uw \in E(AQ_n)$。因为$(F_1,F_2)$并不满足引理\ref{MM*}中的条件，$u$是$F_1 - F_2$中唯一顶点使得$uw \in E(AQ_n)$。如果$F_1 - F_2 = \emptyset$，那么$F_2 \subseteq F_1$, 所以$AQ_n-(F_1 \cup F_2)=AQ_n-F_1$，并且$w$是$AQ_n-F_1$ 中的一个孤立顶点。这与$F_1$是$AQ_n$中的一个$2$-额外集相矛盾。因此，$F_2 - F_1 \neq \emptyset$。 类似的，我们可以得到存在一个顶点$v\in F_2 - F_1$ 使得$vw_1 \in E(AQ_n)$。 当从$AQ_n$中删除$F_2 \cap F_1$时，因为$N(V(H))\cap W = \emptyset$和$N(V(H))\cap (F_1 \cap F_2) = \emptyset$，$AQ_n - F_1 \cap F_2$是不连通的。而且，我们还可以知道$AQ_n-(F_1 \cap F_2)$ 包含一个连通分量$Y$且$|V(Y)| \geq 4$，如图\ref{component>=4}。根据$|V(Y)|$ 的大小，我们有以下两种情况：

\textbf{情形1.} $4 \leq |V(Y)| \leq 6$。因为$N(V(Y)) \subseteq F_1 \cap F_2$，根据引理\ref{k-nodes}，有$|F_1 \cap F_2| \geq |N(V(Y))| \geq 6n-15$。 因为$F_2 \cap F_1 \neq \emptyset$，我们有$|F_2|=|F_1 \cap F_2|+|F_2 - F_1| \geq 6n-15+1$，这与$|F_2| \leq 6n-15$相矛盾。因此，$AQ_n-(F_1 \cup F_2)$不包含孤立顶点。

\textbf{情形2.} $|V(Y)| \geq 7$。因为$W$中的顶点在$F_1$和$F_2$中各仅有一个邻居顶点，那么$|F_1-F_2| \geq 3$或者$|F_2-F_1| \geq 3$。 不失一般性，我们设$|F_1-F_2| \geq 3$。 因为$|F_1|=6n-15$，那么$|F_1 \cap F_2| =|F_1|-|F_1-F_2| \leq 6n-18$。 根据定理\ref{6n-18<=}，$AQ_n-F_1 \cap F_2$只包含一个顶点个数大于2的连通分量。但是当$n \geq 8$ 时，$|H|>2^n - (18n-45) > 2$，且$|V(Y)|>2$。 这意味着$AQ_n-F_1 \cap F_2$中包含两个顶点个数大于2 的连通分量，与定理\ref{6n-18<=}矛盾。因此，$AQ_n-(F_1 \cup F_2)$ 不包含孤立顶点。

对于顶点$p \in V(G)-(F_1 \cup F_2)$, 根据断言，$N(p) \cap (V(G)-(F_1 \cup F_2))$。因为$(F_1, F_2)$并不满足引理\ref{MM*} 中的条件，也就是说$p$在$F_1 \Delta F_2$中没有邻居顶点。因为$p $ 是任意选取的，则$E[V(G)-(F_1 \cap F_2),F_1 \Delta F_2]=\emptyset$. 因为$F_1-F_2 \neq \emptyset$ 和$F_2$是$AQ_n$的一个$2$-额外集，则$|F_1-F_2| \geq 3$。因为$F_1$ 和$F_2$是两个不同的$2$- 额外集，$AQ_n-F_1$和$AQ_n-F_2$ 的任意分量的顶点数都大于$3$。当从$AQ_n$中删除$F_1 \cap F_2$时，$AQ_n-(F_1 \cap F_2)$ 是不连通的且其中的每个连通分量的顶点数都大于等于$3$。根据定义\ref{extra diagnosability definition}(2)，$F_1 \cap F_2$是$AQ_n$ 的一个2-额外割。根据定义\ref{extra diagnosability definition}(3)，$|F_1 \cap F_2| \geq 6n-17$。那么, $|F_1|=|F_1 \cap F_2|+|F_1 - F_2| \geq 6n-17+3 =6n-14 > 6n-15$，矛盾。

因此，$t^{\tilde{m}}_2(AQ_n) \geq 6n-15$。
\end{proof}

(5) $AQ_n-S$有三个分支，其中有两个是$K_1$，且第三个包含$2^n-|S|-2$个顶点。

\begin{lem}
\label{two-components}

设$F$是$AQ_n$的一个额外集，其中$|F| \leq 6n-15$, $n \geq 6$。那么, $AQ_n-S$满足以下$2$个条件之一:

(1) $AQ_n-F$是连通的；

(2) $AQ_n-F$包含两个连通分量，其中一个包含一个顶点个数为$3$的连通分量，另一个是包含顶点个数为$2^n-|F|-3$的连通分量。
\end{lem}

\begin{defn}
\cite{Reliability-analysis-of-data-center-networks}
一个系统$G=(V,E)$和由一个故障顶点集在$G$上产生的症状$\sigma$。设$i \subseteq V$，
$0$-比较集$C_0(i)$定义为$C_0(i)=\{k \in V \mid \exists j \in V$使得$\sigma(i,j)_k=0\}$。$G$的$0$-比较子图，记为$T_0(G)$，是$G$的一个子图，定义为$V(T_0(G)) \subseteq V$和$E(T_0(G))=\{(i,j)\in E \mid j \in C_0(i),i \in C_0(j)\}$。
\end{defn}

\begin{lem}
\cite{Reliability-analysis-of-data-center-networks}
\label{vertex-judge-principle}
给定一个最多有$t$个故障顶点的系统$G=(V,E)$和一个在$MM^*$模型下的$G$上的症状$\sigma$。

(1) 对于任意两个顶点$i,j \in V,(i,j) \in E$，其中$j \in C_0(i),i \in C_0(j)$，那么$i$和$j$有相同的状态（故障或无故障）。

(2) 对于任意一个连通分支$R \subseteq T_0(G)$，那么在$R$中的所有顶点，要么都是故障顶点，要么都是无故障顶点。

(3) 如果$T_0(G)$的连通分支$R$的顶点个数$|V(R)| \geq t+1$，那么$R$中的所有顶点都是无故障的。
\end{lem}


接下来，我们将提出一个$MM^*$模型下$AQ_n$的$2$-额外，$(6n-15)$-可诊断算法。
%该算法基于$AQ_n$ 的两个性质：(1) 对于$AQ_n$中的故障顶点集$|F| \leq 6n-15$，$AQ_n-F$只包含一个顶点个数至少为$2^n-6n-18$的连通分量；(2) $AQ_n$在$MM^*$模型下$AQ_n$的$2$-额外，
%$(6n-15)$-可诊断的。
该算法主要基于引理\ref{two-components}和引理\ref{vertex-judge-principle}，并且这个算法可以正确的诊断出$AQ_n$中的所有的故障顶点。对于一个扩展立方体，我们使用邻接矩阵去存储扩展立方体的结构。首先，我们先给出一个通过深度优先可以确定$AQ_n$中具有同一症状（顶点都为故障或者无故障）的连通分量。

\begin{breakablealgorithm}
\footnotesize
\label{AQ-DFS-MM*}
\caption{\texttt{{AQ-DFS-MM*}}}
\begin{algorithmic}[1]
    \Require 故障顶点集$F$在$G$上生成的症状$\sigma$，其中$|F| \leq 6n-15, n \geq 8$，以及两个无故障的顶点$i,k \in V(AQ_n), (i,k)\in E(AQ_n)$。
    \Ensure  集合$T$。
   \State $T \leftarrow \{i,k\}$;
   \Function{\texttt{DFS-MM*}}{$AQ_n,i,k$}
   \For{$j \in N_{AQ_n}(k)- T$}
        \If{$\sigma(i,j)_k=0$}
            \State $T \leftarrow T \cup \{j\}$;
            \State \texttt{DFS-MM*}($AQ_n,k,j$);
        \EndIf
   \EndFor\\
   \Return $T$
   \EndFunction
  \end{algorithmic}
\end{breakablealgorithm}

算法AQ-DFS-MM*的输入是两个无故障顶点$i,k$且$(i,k) \in E(AQ_n)$，然后根据引理
\ref{vertex-judge-principle}来得到$T_0(G)$。主要通过递归的深度优先算法来实现，因此算法AQ-DFS-MM*的时间复杂度为$O(N^2)$，其中$N$为扩展立方体的顶点个数。

\begin{breakablealgorithm}
\footnotesize
\label{AQ-EXTRA-DIAG}
\caption{\texttt{{AQ-EXTRA-DIAG}}}
\begin{algorithmic}[1]
    \Require 故障顶点集$F$在$G$上生成的症状$\sigma$，其中$|F| \leq 6n-15, n \geq 8$。
    \Ensure  集合$T,F',H$。
    \State $T \leftarrow \emptyset, F' \leftarrow \emptyset, R\leftarrow V(AQ_n)$;
   \State 在$R$中选择$2$个无故障顶点$i,k$，其中$i,k \in E(AQ_n)$;
   \State $T \leftarrow \texttt{DFS-MM*}(AQ_n,i,k)$;
   \If{$|T| \leq 3$}
        %\State $F' \leftarrow N_{AQ_n}(T)$;
        \State 重新选择$2$个无故障顶点$u,v$，其中$u,v \in E(AQ_n)$;
        %\State $T \leftarrow \emptyset$;
        \State $T \leftarrow T \cup \texttt{DFS-MM*}(AQ_n,u,v)$;
        \State $F' \leftarrow R-T$;\\
        \Return $T,F'$;
   \EndIf
   \State
   \State $R \leftarrow R-T$;
   \For{对于顶点$x \in N_{AQ_n}(T) \cap R,$，$y,r \in T$，其中$(x,r) \in E(AQ_n)$和$(y,r)\in E(AQ_n)$}
            \If{$\sigma(x,y)_r=1$}
                \State $F'=F' \cup \{x\}$;
            \EndIf
        \EndFor
   \State $R=R - F'$;
   \State $T \leftarrow T \cup R$;\\
   \Return $T,F'$;
  \end{algorithmic}
\end{breakablealgorithm}

因为算法AQ-EXTRA-DIAG主要以引理\ref{two-components}为基础，因此该算法主要分为$2$种情况：（1）$AQ_n-F$是不连通的，根据引理\ref{two-components}，且在第一次调用算法AQ-DFS-MM时，选取的是小的无故障连通分量$C$且$|C|=3$，调用算法AQ-DFS-MM可以将$C$的所有无故障顶点都诊断出来，然后再次调用算法AQ-DFS-MM，这次可以得到另一个大的无故障连通分量$H$，且$H=2^n-|F|-3$。 这样我们就将所有的无故障顶点的诊断出来，因此剩下的就全部都是故障顶点；（2）我们通过调用算法AQ-DFS-MM得到一个连通分量$H$，其顶点个数大于$3$，根据引理\ref{two-components}，$AQ_n-F$只存在$1$个无故障顶点个数大于$3$的连通分量，因此剩下最多只有$3$个无故障顶点。我们通过与$H$相邻的顶点来判断其是否是故障顶点，如果不是故障顶点，则该顶点属于无故障顶点。这样我们可以将所有故障顶点和无故障顶点诊断出来。

因为算法AQ-EXTRA-DIAG的时间复杂度主要消耗在调用算法AQ-DFS-MM*上，且算法AQ-EXTRA-DIAG 的第$12-16$行的最大时间复杂度也是$O(N^2)$。所以算法AQ-EXTRA-DIAG的时间复杂度是$O(N^2)$，其中$N$为扩展立方体的顶点个数。

\section{本章小结}

在本章中，我们研究了扩展立方体的$2$-额外连通度和$2$-额外诊断度。我们得到的
结果为：（1）扩展立方体的$2$-额外连通度为$\kappa_2(AQ_2)=6n-17$；（2）扩展立方体的的$2$-额外诊断度为$t^{\tilde{m}}_2(AQ_n) = 6n-15$。同时，我们提出了$2$-额外诊断度下的诊断算法AQ-EXTRA-DIAG，实验结果表明，该诊断算法可以正确的诊断出所有的故障顶点。此外，该算法具有通用性，其并不局限于具体的网络拓扑结构，因此算法AQ-EXTRA-DIAG具有一定的应用意义。
