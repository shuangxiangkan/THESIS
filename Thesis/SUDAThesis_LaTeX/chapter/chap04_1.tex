\chapter{广义超立方体的外连通度}
\label{Chap04}
在互连网络中，除了限制连通度以外，
外连通度是衡量网络容错性的另外一个参数。
本章针对广义超立方体，在传统连通度的基础上，
给出广义超体方体上的$1$-外连通度和$2$-外连通度，
并且给出$2$-外连通度条件下的容错路由算法。

本章主要研究广义超立方体的$2$-外连通度，主要结论如下：
(1)对于广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$，
其中$r \geq 5$且对于每个$i \in \langle r \rangle \setminus \{0\}$都有$m_{i}\geq 3$
且$m_{i}\geq m_{i-1}$。
当$m_{r}>m_{r-1}$,
广义超立方体的$2$-外连通度为$3\kappa(G) - 2m_{r}$；
当$m_{r}=m_{r-1}$,
广义超立方体的$2$-外连通度为$3\kappa(G) - 2m_{r}-1$。
(2)给出一个容错路由算法，其时间复杂度为$O(\kappa(G)^{3})$。
\vspace{-1pt}

\section{广义超立方体的$1$-外连通度和$2$-外连通度}

本节首先提出广义超立方体的一些基本性质，
然后分别给出广义超立方体的$1$-外连通度和$2$-外连通度的结果以及其详细的证明过程。

\subsection{广义超立方体的$1$-外连通度}

$1$-外连通度的定义为使$G$不连通，
且使$G$去掉故障顶点后的每个分支中至少有$2$个无故障顶点，
所需要去掉的最少顶点的数目，用$\kappa_{1}(G)$表示。
我们分析可得，下面两个条件是等价的：\\
    (a) 图$G$中去掉故障顶点后，每个无故障顶点都至少包含一个无故障邻接点； \\
    (b) 图$G$中去掉故障顶点后的每个分支中至少有$2$个无故障顶点。\\
因此，关于广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$的$1$-外连通度，
我们可以得到$\kappa_{1}(G)= \kappa^{1}(G)$。
则根据定理\ref{smallerthan}，
我们可以得到$\kappa_{1}(G)=2\kappa(G)-n$，
其中$n=\max\{m_{1},m_{2},\ldots,m_{r}\}$。
如果一个故障的广义超立方体满足故障顶点数$|F|<\kappa_{1}(G)$
且$G-F$的每个分支中至少有$2$个无故障顶点时，
第三章所提出的容错路由算法GHCFP同样适用。

\subsection{广义超立方体上$2$-外连通度下的基本性质}
%定理\ref{minNeighbors}，\ref{oneComponentConnectsToOthers}，\ref{oneComponentConnectsToOthers2}，
%\ref{oneComponentConnectsone}和定理 \ref{oneComponentConnectsone2}及其证明。

关于广义超立方体的$2$-限制连通度，
我们得出结论：
在$G(m_{r},m_{r-1},\ldots,m_{1})$中，
$2$-限制连通度：
\begin{align*}
        \;  &\kappa_{2}(G)=
        \left\{\begin{array}{ll}
            3\kappa(G)-2m_{r}， \mbox{  若 }m_{r}>m_{r-1} \\
            3\kappa(G)-2m_{r}-1， \mbox{ 若 }m_{r}=m_{r-1} \\
        \end{array}\right.
\end{align*}

在给出广义超立方体的$2$-外连通度的具体证明之前，
我们在本小节给出广义超立方体的一些基本的性质。

\begin{lem}
\label{minNeighbors}
在$G(m_{r},m_{r-1},\ldots,m_{1})$中，
令$u,v,w$为子图$G^k[j]$中的$3$个顶点，
其中$(u,v)\in E(G^k[j])$并且$(u,w) \in E(G^k[j])$，
那么我们可以得到：
\begin{align*}
        \;  &|N_{G^k[j]}(u,v,w)|\geq
        \left\{\begin{array}{ll}
            3\kappa(G)-3m_{k}-2m_{r}+3， \mbox{ 若 }m_{r}>m_{r-1} \\
            3\kappa(G)-3m_{k}-2m_{r}+2， \mbox{ 若 }m_{r}=m_{r-1}  \\
        \end{array}\right.
\end{align*}

\end{lem}

\begin{proof}
我们令$u=u_{r}u_{r-1}\dots u_{1}$，$v=v_{r}v_{r-1}\dots v_{1}$
和$w=w_{r}w_{r-1}\dots w_{1}$。
根据定义\ref{GeneralizedHypercube}，
我们可以得到存在唯一一个整数$o$使得$u_{o}\neq v_{o}$
且对于任意一个$i \in \langle r \rangle \setminus \{o,0\}$，都有$u_{i}=v_{i}$。
然后，我们可以进一步得到
$N_{G^k[j]}(u)\cap N_{G^k[j]}(v)=\{y_{r}y_{r-1}\dots y_{1}\mid
y_{o}\in \langle m_{o}-1 \rangle \setminus \{u_{o},v_{o}\}$,
对于任意一个$i \neq o$都有$y_{i}=u_{i}=v_{i}\}$。
显然, $|N_{G^k[j]}(u)\cap N_{G^k[j]}(v)| = m_{o}-2$。
同理, 对于$u$和$v$，我们可以得到有唯一一个整数$p$使得$u_{p}\neq w_{p}$，
并且对于每一个$i \in \langle r \rangle \setminus \{p,0\}$都有$u_{i}=u_{i}$。
同理，我们可以得到$|N_{G^k[j]}(v)\cap N_{G^k[j]}(w)| = m_{p}-2$。
因为$G^k[j]$中含有长度为$3$的环。那么，我们讨论以下$2$种情形。

情形1. $(v,w) \in E(G^k[j])$。

不失一般性，我们令$v$和$w$在第$q$维不同，
也就是$v_{q}\neq w_{q}$并且对于任意一个$i \in \langle r \rangle \setminus \{q,0\}$，
都有$v_{i}=w_{i}$。
类似地，我们有$|N_{G^k[j]}(v)\cap N_{G^k[j]}(w)| = m_{q}-2$。
%并且，可以得到$o=p=q$。
%证明如下，相反地，
如果我们假设$o \neq p$。
因为$u_{o} \neq v_{o}$且$u_{p} \neq w_{p}$，
所以，我们可以得到$v_{o} \neq w_{o}$和$v_{p} \neq w_{p}$。
该结论和$v$和$w$仅在第$q$维不同相矛盾。
因此，假设$o \neq p$不成立。
同理，我们可以得到$o=q$。故$m_{o}=m_{p}=m_{q}$。
同时，因为$N_{G^k[j]}(u)\cap N_{G^k[j]}(v)\cap N_{G^k[j]}(w)
=\{y_{r}y_{r-1}\dots y_{1}|
y_{o}\in \langle m_{o}-1 \rangle \setminus \{u_{o},v_{o},w_{o}\}$且对于任意一个 $i \neq o$都有$y_{i}=u_{i}=v_{i}\}$。
所以，我们可以得到$|N_{G^k[j]}(u)\cap N_{G^k[j]}(v)\cap N_{G^k[j]}(w)|= m_{o}-3$。
因此，$|N_{G^k[j]}(u,v,w)|
=|N_{G^k[j]}(u)| + |N_{G^k[j]}(v)| + |N_{G^k[j]}(w)|
- |N_{G^k[j]}(u)\cap N_{G^k[j]}(v)| -|N_{G^k[j]}(u)\cap N_{G^k[j]}(w)|-|N_{G^k[j]}(v)\cap N_{G^k[j]}(w)|
+ |N_{G^k[j]}(u)\cap N_{G^k[j]}(v)\cap N_{G^k[j]}(w)|
- |\{u,v,w\}|
= 3(\kappa(G)-(m_{k}-1))-(m_{o}-2)-(m_{p}-2)-(m_{q}-2)+(m_{o}-3)-3
= 3\kappa(G)-3m_{k}-2m_{o}+3$。
更进一步，因为$m_{o}\leq m_{k}$，
可以得到$|N_{G^k[j]}(u,v,w)|\geq 3\kappa(G)-3m_{k}-2m_{k}+3$。

情形 2. $(v,w) \notin E(G^k[j])$。

既然$u_{o} \neq v_{o}$，并且$u_{p} \neq w_{p}$，
那么我们可以得到$v_{o} \neq w_{o}$和$v_{p} \neq w_{p}$。
因此，有$|N_{G^k[j]}(v)\cap N_{G^k[j]}(w)|=2$
且$|N_{G^k[j]}(u)\cap N_{G^k[j]}(v)\cap N_{G^k[j]}(w)| = 0$。
所以有$|N_{G^k[j]}(u,v,w)|
= 3(\kappa(G)-(m_{k}-1))-(m_{o}-2)-(m_{p}-2)-2+0-3
= 3\kappa(G)-3m_{k}-m_{o}-m_{p}+2$。

如果$m_{r}>m_{r-1}$，
有$m_{o}+m_{p} < 2m_{r}$。
因此，$|N_{G^k[j]}(u,v,w)|> 3\kappa(G)-3m_{k}-2m_{r}+2$；
如果 $m_{r}=m_{r-1}$，
我们有$m_{o}+m_{p} \leq 2m_{r}$。
因此，$|N_{G^k[j]}(u,v,w)| \geq 3\kappa(G)-3m_{k}-2m_{r}+2$。
综上所述，定理成立。
\end{proof}

\begin{defn}
  \label{cn-number}
对于图$G$来说，我们定义$cn_{G}(S_{1},S_{2})=|N_{G}(S_1)\cap N_{G}(S_1)|$，
其中$V(S_{1})\subset V(G)$, $V(S_{2})\subset V(G), V(S_{1})\cap V(S_{2})=\emptyset$并且
$S_{1} \not \subset N_{G}(S_{2})$。
\end{defn}

%induction method on $n$.
根据定义\ref{GeneralizedHypercube}，我们很容易可以得到引理\ref{cn1}。

\begin{lem}
\label{cn1}
在图$G(m_{r},m_{r-1},\ldots,m_{1})$中，其中$r\geq1$并且对于任意一个$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i}\geq 3$，
当$S_{1} \cong K_{1}$且$S_{2} \cong K_{1}$，
则$cn_{G}(S_{1},S_{2})\leq 2$。
\end{lem}

基于引理\ref{cn1}，我们可以扩展出以下引理。

\begin{lem}
\label{cn2}
在图$G(m_{r},m_{r-1},\ldots,m_{1})$中，其中$r\geq1$并且对于任意一个$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i}\geq 3$，
当$S_{1} \cong K_{1}$且$S_{2} \cong P_{2}$，
有$cn_{G}(S_{1},S_{2}) \leq 3$。
\end{lem}

\begin{proof}
%我们根据$r$进行数学归纳。

首先，对于$G(m_{r},m_{r-1},\ldots,m_{1})$，
当$r=1$时，
该引理显然成立。
接下来，我们证明该引理在$r\geq2$时成立。
令$S_1=\{u\}$且$S_2=\{x,y\}$，满足$(x,y)\in E(G)$。
其中$(x,y)$是第$j$-维的边。
我们从第$j$-维将$G$切分为$m_{j}$个子图
$G^{j}[0],G^{j}[1],...,G^{j}[m_{j}-1]$。
此时很显然$x$和$y$不属于同一个子图。
我们假设$u \in G^{j}[i]$，其中$i \in \langle m_{j}-1 \rangle$。
根据$x$和$y$的不同位置，我们分析以下两种情形。

情形1. $x \in G^{j}[i]$ 或者 $y \in G^{j}[i]$。

不失一般性，我们假设$x \in G^{j}[i]$，并且$y \in G^{j}[i+1]$。
根据引理\ref{cn1}，我们有$cn_{G^{j}[i]}(\{u\},\{x\}) \leq 2$(如图\ref{cn2_figure}(a))。
然后，显然我们有$cn_{G-G^{j}[i]}(\{u\},\{x\}) =0$。
同时，根据广义超立方体的定义，
我们有$cn_{G^{j}[i+1]}(\{u\},\{y\}) \leq 1$。
然后，我们假设$N_{G-G^{j}[i+1]}(u)\cap N_{G-G^{j}[i+1]}(y)\neq \emptyset$，
那么我们有$(u,y) \in E(G)$，
这和$\{u\} \not \subset N_{G}(y)$相矛盾。
所以，我们有$N_{G-G^{j}[i+1]}(u)\cap N_{G-G^{j}[i+1]}(y)=\emptyset$，
也就是$cn_{G-G^{j}[i+1]}(\{u\},\{y\}) =0$。
因此，我们有
$cn_{G}(S_1,S_2)=
cn_{G^{j}[i]}(\{u\},\{x\})+
cn_{G-G^{j}[i]}(\{u\},\{x\}) +
cn_{G^{j}[i+1]}(\{u\},\{y\})+
cn_{G-G^{j}[i+1]}(\{u\},\{y\})
\leq 2+0+1+0=3$。

情形2. $x \notin G^{j}[i]$ 并且 $x \notin G^{j}[i]$。

我们假设$x \in G^{j}[i+1]$，并且$y \in G^{j}[i+2]$。
那么根据广义超立方体的定义，我们有
$cn_{G^{j}[i]}(\{u\},\{x,y\}) \leq 1$，
同时，我们有
$cn_{G^{j}[i+1]}(\{u\},\{x\}) \leq 1$
且$cn_{G^{j}[i+2]}(\{u\},\{y\}) \leq 1$
(如图\ref{cn2_figure}(b))。
然后，我们假设$N_{G-(G^{j}[i]\cup G^{j}[i+1])}(u)\cap N_{G-(G^{j}[i]\cup G^{j}[i+1])}\\
(x)\neq \emptyset$，
那么我们有$(u,x) \in E(G)$，
这和$\{u\} \not \subset N_{G}(x)$相矛盾。
所以，我们有$N_{G-(G^{j}[i]\cup G^{j}[i+1])}\\
(u)\cap N_{G-(G^{j}[i]\cup G^{j}[i+1])}(x)=\emptyset$，
也就是$cn_{G-(G^{j}[i]\cup G^{j}[i+1])}(\{u\},\{x\}) =0$。
同理，我们还可以得到
$cn_{G-(G^{j}[i]\cup G^{j}[i+2])}
(\{u\},\{y\}) =0$。
所以，我们有$cn_{G}(S_{1},S_{2})
=cn_{G^{j}[i]}(\{u\},\{x,y\})+
cn_{G^{j}[i+1]}(\{u\},\{x\})+
cn_{G^{j}[i+1]}(\{u\},\{y\})+
cn_{G-(G^{j}[i]\cup G^{j}[i+1])}(\{u\},\{x\})+
cn_{G-(G^{j}[i]\cup G^{j}[i+2])}(\{u\},\{y\})
\leq 1+1+1+0+0=3$。

综上所述，我们有$cn_{G}(S_{1},S_{2}) \leq 3$，该引理成立。
\end{proof}

\begin{lem}
\label{cn3}
在图$G(m_{r},m_{r-1},\ldots,m_{1})$中，其中$r\geq1$并且对于任意一个$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i}\geq 3$，
那么，当$S_{1} \cong K_{1}$且$S_{2} \cong P_{3}$，
有$cn_{G}(S_{1},S_{2}) \leq 4$。
\end{lem}

\begin{proof}
首先，对于$G(m_{r},m_{r-1},\ldots,m_{1})$，
当$r=1$时，
该引理显然成立。
接下来，我们证明该引理在$r\geq2$时成立。
令$S_1=\{u\}$且$S_2=\{x,y,z\}$，
其中$(x,y),(y,z) \subseteq E(G)$。
假设$(x,y)$是第$j$-维的边，
那么我们从第$j$-维将$G$切分为$m_{j}$个子图
$G^{j}[0],G^{j}[1],...,G^{j}[m_{j}-1]$。
此时很显然$x$和$y$不属于同一个子图。
我们假设$u \in G^{j}[i]$，其中$i \in \langle m_{j}-1 \rangle$。
根据$x$,$y$和$z$的不同位置，我们分析以下两种情形。

情形1.$z$和$y$在同一个子图。

情形1.1. $x \in G^{j}[i]$且$y \in G^{j}[i+1]$。

根据引理\ref{cn1}，我们有$cn_{G^{j}[i]}(\{u\},\{x\}) \leq 2$。
%并且因为$\{u\} \not \subset N_{G}(\{x\})$，
%容易得到$cn_{G\setminus G^{j}[i]}(\{u\},\{x\})=0$。
同时，根据广义超立方体的定义，我们可以得到
$cn_{G^{j}[i]}(\{u\},\{y,z\})=cn_{G^{j}[i]}(\{u\},\{z\})\leq 1$，
同时$cn_{G^{j}[i+1]}(\{u\},\{y,z\})\leq 1$(如图\ref{ch4_lemma2_1}(a))。
然后，我们假设$N_{G-G^{j}[i]}(u)\cap N_{G-G^{j}[i]}(x)\neq \emptyset$，
那么我们有$(u,x) \in E(G)$，
这和$\{u\} \not \subset N_{G}(x)$相矛盾。
所以，我们有$N_{G-G^{j}[i]}(u)\cap N_{G-G^{j}[i]}(x)=\emptyset$，
也就是$cn_{G-G^{j}[i]}(\{u\},\{x\}) =0$。
同理，我们有
$cn_{G-(G^{j}[i] \cup G^{j}[i+1])}(\{u\},\{y,z\}) =0$。
所以，我们有$cn_{G}(S_{1},S_{2})=
cn_{G^{j}[i]}(\{u\},\{x\})+
cn_{G^{j}[i]}(\{u\},\{y,z\})+
cn_{G^{j}[i+1]}(\{u\},\{y,z\})+
cn_{G-G^{j}[i]}(\{u\},\{x\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+1])}(\{u\},\{y,z\})
\leq 2+1+1+0+0=4$。

情形1.2. $x \in G^{j}[i+1]$且$y \in G^{j}[i+2]$。

根据定义\ref{GeneralizedHypercube}，
我们有$cn_{G^{j}[i]}(\{u\},\{x\})=cn_{G^{j}[i]}(\{u\},\{y\}) \leq 1$，
$cn_{G^{j}[i+1]}\\
(\{u\},\{x\}) \leq 1$，
$cn_{G^{j}[i]}(\{u\},\{z\}) \leq 1$，
且$cn_{G^{j}[i+2]}(\{u\},\{y,z\}) \leq 1$(如图\ref{ch4_lemma2_1}(b))。
然后，我们假设$N_{G-(G^{j}[i]\cup G^{j}[i+1])}(u)\cap
N_{G-(G^{j}[i]\cup G^{j}[i+1])}(x)\neq \emptyset$，
那么我们有$(u,x) \in E(G)$，
这和$\{u\} \not \subset N_{G}(x)$相矛盾。
所以，我们有$N_{G-(G^{j}[i]\cup G^{j}[i+1])}(u)\cap N_{G-(G^{j}[i]\cup G^{j}[i+1])}(x)=\emptyset$，
也就是$cn_{G-(G^{j}[i]\cup G^{j}[i+1])}(\{u\},\{x\}) =0$。
同理，我们有
$cn_{G-(G^{j}[i]\cup G^{j}[i+2])}(\{u\},\{y,z\}) =0$。
所以，我们有$cn_{G}(S_{1},S_{2})=
cn_{G^{j}[i]}(\{u\},\{x\})+
cn_{G^{j}[i+1]}(\{u\},\{x\})+
cn_{G^{j}[i]}(\{u\},\{z\})+
cn_{G^{j}[i+2]}(\{u\},\{y,z\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+1])}(\{u\},\{x\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+2])}(\{u\},\{y,z\})
\leq 1+1+1+1+0+0=4$。

情形1.3. $x \in G^{j}[i+1]$且$y \in G^{j}[i]$。

根据引理\ref{cn2}，我们有$cn_{G^{j}[i]}(\{u\},\{y,z\}) \leq 3$。
同时，我们有$cn_{G^{j}[i+1]}(\{u\},\{x\}) \leq 1$。
然后，我们假设$N_{G-G^{j}[i+1]}(u)\cap N_{G-G^{j}[i+1]}(x)\neq \emptyset$，
那么我们有$(u,x) \in E(G)$，
这和$\{u\} \not \subset N_{G}(x)$相矛盾。
所以，我们有$N_{G-G^{j}[i+1]}(u)\cap N_{G-G^{j}[i+1]}(x)=\emptyset$，
也就是$cn_{G-G^{j}[i+1]}(\{u\},\{x\}) =0$。
同理，我们有
$cn_{G-G^{j}[i]}(\{u\},\{y,z\}) =0$。
所以，我们有$cn_{G}(S_{1},S_{2})=
cn_{G^{j}[i]}(\{u\},\{y,z\})+
cn_{G-(G^{j}[i]}(\{u\},\{y,z\})+
cn_{G^{j}[i+1]}(\{u\},\{x\})+
cn_{G-(G^{j}[i+1]}(\{u\},\{x\})
\leq 3+0+1+0=4$。

情形2. $z$和$y$不在同一个子图。

情形2.1. $x \in G^{j}[i-1], y \in G^{j}[i]$ 且 $y \in G^{j}[i+1]$。

根据广义超立方体的定义，我们有$cn_{G^{j}[i]}(\{u\},\{y\}) \leq 2$，
$cn_{G^{j}[i-1]}(\{u\},\{x\}) \leq 1$，
且$cn_{G^{j}[i+1]}(\{u\},\{z\})\leq 1$(如图\ref{ch4_lemma2_2}(a))。
然后，我们假设$N_{G-G^{j}[i]}(u)\cap N_{G-G^{j}[i]}(y)\neq \emptyset$，
那么我们有$(u,y) \in E(G)$，
这和$\{u\} \not \subset N_{G}(y)$相矛盾。
所以，我们有$N_{G-G^{j}[i+1]}(u)\cap N_{G-G^{j}[i+1]}(y)=\emptyset$，
也就是$cn_{G-G^{j}[i]}(\{u\},\{y\}) =0$。
同理，我们有
$cn_{G-G^{j}[i-1]}(\{u\},\{x\}) =0$
和$cn_{G-G^{j}[i+1]}(\{u\},\{z\}) \\
=0$。
所以，我们有$cn_{G}(S_{1},S_{2})=
cn_{G^{j}[i]}(\{u\},\{y\})+
cn_{G-(G^{j}[i]}(\{u\},\{y\})+
cn_{G^{j}[i-1]}(\{u\},\{x\})+
cn_{G-(G^{j}[i-1]}(\{u\},\{x\})+
cn_{G^{j}[i+1]}(\{u\},\{z\})+
cn_{G-(G^{j}[i+1]}(\{u\},\{z\})
\leq 2+0+1+0+1=4$。

情形2.2. $x \in G^{j}[i+1], y \in G^{j}[i+2]$且$y \in G^{j}[i+3]$。

需要注意的是，该情况只有在$m_j\geq4$时出现。
根据广义超立方体的定义，
我们有$cn_{G^{j}[i+1]}(\{u\},\{x,y,z\})\leq 1$，
$cn_{G^{j}[i+1]}(\{u\},\{x\}) \leq 1$，
$cn_{G^{j}[i+2]}(\{u\},\{y\}) \leq 1$q且
$cn_{G^{j}[i+3]}(\{u\},\{z\})\\
\leq 1$(如图\ref{ch4_lemma2_2}(b))。
然后，我们假设$N_{G-(G^{j}[i]\cup G^{j}[i+1])}(u)\cap N_{G-(G^{j}[i]\cup G^{j}[i+1])}
(x)\neq \emptyset$，
那么我们有$(u,x) \in E(G)$，
这和$\{u\} \not \subset N_{G}(x)$相矛盾。
所以，我们有$N_{G-(G^{j}[i]\cup G^{j}[i+1])}(u)\cap N_{G-(G^{j}[i]\cup G^{j}[i+1])}\\
(x)=\emptyset$，
也就是$cn_{G-(G^{j}[i]\cup G^{j}[i+1])}(\{u\},\{x\}) =0$。
同理，我们有
$cn_{G-(G^{j}[i]\cup G^{j}[i+2])}(\{u\},\{y\}) =0$
且$cn_{G-(G^{j}[i]\cup G^{j}[i+2])}(\{u\},\{z\}) =0$。
所以，我们有$cn_{G}(S_{1},S_{2})=
cn_{G^{j}[i+1]}(\{u\},\{x,y,z\})+
cn_{G^{j}[i+1]}(\{u\},\{x\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+1])}(\{u\},\{x\})+
cn_{G^{j}[i+2]}(\{u\},\{y\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+2])}(\{u\},\{y\})+
cn_{G^{j}[i+3]}(\{u\},\{z\})+
cn_{G-(G^{j}[i] \cup G^{j}[i+3])}(\{u\},\{z\})
\leq 1+1+0+1+0+1+0=4$。

\begin{lem}
\label{cn4}
在图$G(m_{r},m_{r-1},\ldots,m_{1})$中，其中$r\geq1$并且对于任意一个$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i}\geq 3$，
那么，当$S_{1} \cong K_{2}$且$S_{2} \cong P_{3}$，
有$cn_{G}(S_{1},S_{2}) \leq 8$。
\end{lem}

\begin{proof}
令$V(S_1)=\{u,v\}$，$S_2=\{x,y,z\}$，
其中，$(u,v),(x,y),(y,z)\in E(G)$。
根据引理\ref{cn3}，$cn_{G}(S_{1},S_{2})
\leq cn_{G}(\{u\},S_{2})+
cn_{G}(\{v\},S_{2})
=4+4=8$。
\end{proof}

\begin{lem}
\label{oneComponentConnectsToOthers}
令$F$为广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$中的一个故障点集合，
其中当$m_{r}>m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-1$；
当$m_{r}=m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-2$。
如果$G-F$中的每个分支中都包含至少$3$个顶点，
那么对于每个$j \in \langle m_{k}-1 \rangle$，
都有子图$G^{k}[j]-F_{j}$中每个顶点可以连通到子图$G^{k}[J]-F_{J}$,
其中$J=\langle m_{k}-1 \rangle \setminus \{j\}$。
\end{lem}

\begin{proof}
令顶点$u\in V(G^{k}[j]-F_{j})$，
且$N_{G^{k}[J]}(u)$为顶点$u$在子图$G^{k}[J]$中的邻居集合。
如果存在一个顶点$u'$满足$u' \in N_{G^{k}[J]}(u)- F$，
那么本定理得证；
否则，因为$G-F$不存在孤立点，且$N_{G^{k}[J]}(u)\subseteq F$，
所以我们可以得到存在一个顶点$v \in N_{G^{k}[j]}(u)-F_{j}$。
如果$N_{G^{k}[J]}(v)\not\subseteq F_{J}$，本定理得证；
否则，因为$G-F$中不存在孤立边，
$N_{G^{k}[J]}(u)\subseteq F$且$N_{G^{k}[J]}(v)\subseteq F$，
所以存在一个顶点$w \in N_{G^{k}[j]}(\{u,v\})-F_{j}$。
如果$N_{G^{k}[J]}(w)\not\subseteq F_{J}$，本定理得证；
否则，也就是$N_{G^{k}[J]}(w)\subseteq F_{J}$。
那么顶点$w$可能和顶点$u$或者$v$相邻。
不失一般性，我们假设$w$和$v$相邻。
我们考虑以下两种情形。

情形1. $m_{r}>m_{r-1}$

根据引理~\ref{minNeighbors}，$|N_{G^k[j]}(\{u,v,w\})|\geq 3\kappa(G)-3m_{k}-2m_{r}+3$。
令$N=N_{G^k[j]}(\{u,v,w\})\cup N_{G^k[J]}\{u,v,w\}$。
那么，$|N|\geq (3\kappa(G)-3m_{k}-2m_{r}+3)+3(m_{k}-1)=3\kappa(G)-2m_{r}$。
相反地，我们假设$N$中的任何一个顶点都无法和子图$G^{k}[J]$的顶点连通。
那么我们有$3\kappa(G)-2m_{r}-1\geq|F|\geq |N|$。
这和$|N|\geq3\kappa(G)-2m_{r}>|F|$相矛盾。
因此，存在一个顶点可以和子图$G^{k}[J]$
中的顶点相连。

情形2. $m_{r}=m_{r-1}$

根据引理~\ref{minNeighbors}，$|N_{G^k[j]}(\{u,v,w\})|\geq 3\kappa(G)-3m_{k}-2m_{r}+2$。
令$N=N_{G^k[j]}(\{u,v,w\})\cup N_{G^k[J]}\{u,v,w\}$。
那么，$|N|\geq (3\kappa(G)-3m_{k}-2m_{r}+2)+3(m_{k}-1)=3\kappa(G)-2m_{r}-1$。
相反地，我们假设$N$中的任何一个顶点都无法和子图$G^{k}[J]$的顶点连通。
那么我们有$3\kappa(G)-2m_{r}-2\geq|F|\geq |N|$。
这和$|N|\geq3\kappa(G)-2m_{r}-1>|F|$相矛盾。
因此，存在一个顶点可以和子图$G^{k}[J]$
中的顶点相连。

综上所述，子图$G^{k}[j]-F_{j}$中顶点$u$可以连通到子图$G^{k}[J]-F_{J}$,得证。
\end{proof}

\begin{lem}
\label{oneComponentConnectsone}
令$G^{k}[\alpha]$和$G^{k}[\beta]$为广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$中的两个子图，
$F_{\alpha}$和$F_{\beta}$分别为
$G^{k}[\alpha]$和$G^{k}[\beta]$两个子图中的故障集合，
且$|F_{\alpha}| \leq \kappa^{1}(G^{k}[\alpha])$，
$|F_{\beta}| \leq \kappa^{1}(G^{k}[\beta])$。
若$m_{r}>m_{r-1}$，$|F_{\alpha}| \bigcup F_{\beta}| \leq 3\kappa(G)-3m_{k}-2m_{r}+5$。
若$m_{r}=m_{r-1}$，$|F_{\alpha}| \bigcup F_{\beta}| \leq 3\kappa(G)-3m_{k}-2m_{r}+4$。
如果$G^{k}[\alpha]-F_{\alpha}$和$G^{k}[\beta]-F_{\beta}$中
的每个顶点都至少有一个无故障邻居，
那么$G^{k}[\alpha]-F_{\alpha}$中的每个顶点可以和$G^{k}[\beta]-F_{\beta}$中的某些顶点相连。
\end{lem}


\begin{proof}
令顶点$u$满足$u \in V(G^{k}[\alpha]-F_{\alpha})$，
并且令$u'$为$u$在子图$G^{k}[\beta]$中的邻居。
如果$u' \notin F_{\beta}$，本定理得证；
否则的话，也就是$u' \in F_{\beta}$。
既然$u$在$G^{k}[\alpha]-F_{\alpha}$中不是一个孤立点，
存在一个顶点$v \subset N_{G^{k}[\alpha]-F_{\alpha}}(u)$。
如果$v' \notin F_{\beta}$，本引理得证；
否则话，也就是$v' \in F_{\beta}$。
既然$|F_{\alpha}| \leq \kappa^{1}(G^{k}[\alpha])$，
那么$(u,v)$不是子图$G^{k}[\alpha]-F_{\alpha}$一条孤立边。
存在一个顶点$w \in N_{G^{k}[\alpha]}(u,v)$。
如果其邻居$w' \notin F_{\beta}$，本引理得证；
否则的话，也就是$w' \in F_{\beta}$。
我们可以考虑以下两种情形。

情形1. $m_{r}>m_{r-1}$

根据引理~\ref{minNeighbors}，
我们可以得到
$|N_{G^k[j]}(\{u,v,w\})|\geq 3\kappa(G)-3m_{1}-2m_{r}+3$。
令$N=N_{G^k[\alpha]}(u,v,w)\cup N_{G^k[\beta]}(u,v,w)$。
然后，$|N|\geq (3\kappa(G)-3m_{k}-2m_{r}+3)+3=3\kappa(G)-3m_{k}-2m_{r}+6$。
相反地，我们假设
$N$中的任意一个顶点都无法连接到图$G^{k}[\beta]-F_{\beta}$。
那么，可以得到$|F_{\alpha}\cup F_{\beta}|\geq |N| \geq 3\kappa(G)-3m_{k}-2m_{r}+6$。
这个和$|F_{\alpha}\cup F_{\beta}|\leq 3\kappa(G)-3m_{k}-2m_{r}+5$互相矛盾。
因此，可以得到$N$中存在一个顶点可以连接到子图$G^{k}[\beta]-F_{\beta}$。

情形2. $m_{r}=m_{r-1}$

根据引理~\ref{minNeighbors}，
我们可以得到
$|N_{G^k[j]}(\{u,v,w\})|\geq 3\kappa(G)-3m_{1}-2m_{r}+2$。
令$N=N_{G^k[\alpha]}(u,v,w)\cup N_{G^k[\beta]}(u,v,w)$。
然后，$|N|\geq (3\kappa(G)-3m_{k}-2m_{r}+2)+3=3\kappa(G)-3m_{k}-2m_{r}+5$。
相反地，我们假设
$N$中的任意一个顶点都无法连接到图$G^{k}[\beta]-F_{\beta}$。
那么，可以得到$|F_{\alpha}\cup F_{\beta}|\geq |N| \geq 3\kappa(G)-3m_{k}-2m_{r}+4$。
这个和$|F_{\alpha}\cup F_{\beta}|\leq 3\kappa(G)-3m_{k}-2m_{r}+5$相矛盾。
因此，可以得到$N$中存在一个顶点可以连接到子图$G^{k}[\beta]-F_{\beta}$。

综上所述，$G^{k}[\alpha]-F_{\alpha}$中的顶点$u$可以和$G^{k}[\beta]-F_{\beta}$ 中的某些顶点相连，得证。
\end{proof}

基于引理~\ref{oneComponentConnectsToOthers}和引理~\ref{oneComponentConnectsone}，
我们可以得到如下定理。


\begin{theorem}
  \label{ConnectivityLargerThan}
在广义超立方体$G(m_{r}, m_{r-1}, \ldots, m_{1})$中，
若$r \geq 5$，并且对于所有的$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i} \geq 3$。
如果有一个故障点集合$F \subset V(G(m_{r}, m_{r-1}, \ldots, m_{1}))$
满足当$m_{r}>m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-1$；
当$m_{r}=m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-2$，
并且图$G(m_{r}, m_{r-1}, \ldots, m_{1})-F$的每个分支至少包含
$3$个无故障顶点，
那么图$G(m_{r}, m_{r-1}, \ldots, m_{1})-F $是连通的。
\end{theorem}

\begin{proof}
证明本定理，等价于证明：若$m_{r}>m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-1$；
若$m_{r}=m_{r-1}$有$|F|\leq 3\kappa(G)-2m_{r}-2$，
且$G(m_{r}, m_{r-1}, \ldots, m_{1})-F$中每个分支中至少有一个无故障邻居时，
图$G-F$中任意两个无故障顶点$u$和$v$之间存在一条无故障路径。

若$u$和$v$相邻，则本定理得证。
因此，我们讨论$u$和$v$不相邻。

首先，因为
若$m_{r}>m_{r-1}$且$|F|\leq 3\kappa(G)-2m_{r}-1$，
那么有$\frac{3\kappa(G)-2m_{r}-1}{\kappa(G)-1}\leq 3$；
若$m_{r}=m_{r-1}$且$|F|\leq 3\kappa(G)-2m_{r}-2$，
那么同样有$\frac{3\kappa(G)-2m_{r}-2}{\kappa(G)-1}\leq 3$，
所以可以得知，$G-F$中至多有$3$个顶点其分支度为$1$。
首先，我们假设存在三个顶点的分支度为$1$,
且其关联边分别为第$i,j,l$三个维度的跨边，
其中$\{i,j,l\}\subset \langle r \rangle \setminus \{0\}$
那么，既然$r\geq5$，我们必然能够找到一个维度$k$，
使得$k\subset \langle r \rangle \setminus \{0,i,j,l\}$
并且$m_{k}=$ min$\{\{m_{1},m_{2}\ldots,m_{r}\}\setminus\{m_{i},m_{j},m_{l}\}\}$。
同理，如果分支度为$1$的顶点数为$\{0,1,2\}$，我们可以找到维度$k$。
这样，我们就可以保证对于每个$j\subset \langle m_{k}-1 \rangle$，
都有子图$G^{k}[j]$中不存在孤立点。

假设$u \in G^{r}[\alpha]$且$v \in G^{r}[\beta]$,
其中$\alpha,\beta \in \langle m_{r}-1\rangle$。
同时让$F_{\alpha}=V(G^{r}[\alpha]) \cap F$
且$F_{\beta}=V(G^{r}[\beta])\cap F$。
接下来分为以下两种情形讨论。

情形1. $\alpha \neq \beta$。

情形1.1. $|F_{\alpha}| \geq \kappa^1(G^{k}[\alpha])$或$|F_{\beta}| \geq \kappa^1(G^{k}[\beta])$。

不失一般性，我们假设$|F_{\alpha}| \geq \kappa(G^{r}[\alpha])$。
让$J=\langle m_r-1 \rangle\setminus \{\alpha\}$且
$F_J = \bigcup_{j \in J}F_j$。
则有$G^r[J]=G-V(G^r[\alpha])$。
结合引理~\ref{oneComponentConnectsToOthers}，
我们可以得到，顶点$u$可以通过一条无故障路径连通到$G^k[J]$图中的一个顶点，
我们假设该顶点为$u'$，假设该路径为$P_1$。
接下来，因为$|F_{\alpha}|\geq \kappa^1(G^{k}[\alpha])=2\kappa-2m_{k}-m_r+2$，
如果$m_{r}>m_{r-1}$,
我们有$|F_{J}|=|F-F_{\alpha}|\leq (3\kappa-2m_r-1)-(2\kappa-2m_{k}-m_r+2)
=\kappa-m_r+2m_{k}-3$。
又$\kappa^1(G^k[J])=2(\kappa-1)-m_r=2\kappa-m_r-2$，
我们可以得到$\kappa^1(G^k[J])-|F_{J}|=\kappa-2m_{k}+1$，
显然$\kappa-2m_{k}+1>0$，所以我们有$\kappa^1(G^k[J])>|F_{J}|$。
如果$m_{r}=m_{r-1}$,
同理，
我们有$|F_{J}|=|F-F_{\alpha}|\leq (3\kappa-2m_r-2)-(2\kappa-2m_{k}-m_r+2)
=\kappa-m_r+2m_{k}-4$。
又$\kappa^1(G^k[J])=2(\kappa-1)-m_r=2\kappa-m_r-2$，
我们可以得到$\kappa^1(G^k[J])-|F_{J}|=\kappa-2m_{k}+1$，
显然$\kappa-2m_{k}>0$，所以我们仍然有$\kappa^1(G^k[J])>|F_{J}|$。
又因为$G^k[J]$中每个顶点都至少有一个无故障邻居，
我们可以得到$G^r[J]-F_{J}=G-V(G^r[\alpha])-F$是连通的。
则在图$G^r[J]-F_{J}$中$u'$和$v$之间存在一条无故障路径$P_2$。
因此，在$G-F$中，路径$(P_1,P_2)$是$u$到$v$的一条无故障路径(如图\ref{example1_1})。

情形1.2. $|F_{\alpha}| < \kappa^1(G^{k}[\alpha])$ 且 $|F_{\beta}| < \kappa^1(G^{k}[\beta])$。

因为$|F_{\beta}| < \kappa^1(G^{k}[\beta])$，
则根据$1$-限制连通度的定义，子图$G^k[\beta]-F_{\beta}$中
不存在孤立点和孤立边，且子图$G^k[\beta]-F_{\beta}$是连通的。
令$x$是$u$在子图$G^k[\beta]$中的邻居。
若$x \notin F$，既然$G^r[\beta]-F_{\beta}$是连通的，
那么子图$G^k[\beta]-F_{\beta}$中的$x$和$v$之间存在一条无故障路径$P$。
则我们在$G-F$找到从$u$到$v$之间的一条无故障路径$(x,P)$。
否则的话，若$x \in F$，我们分以下两种情况讨论。

情形1.2.1. $m_{r}>m_{r-1}$且$|F_{\alpha}\cup F_{\beta}|\leq 3\kappa(G)-2m_k-2m_r+5$
或$m_{r}=m_{r-1}$且$|F_{\alpha}\cup F_{\beta}|\leq 3\kappa(G)-2m_k-2m_r+4$。

既然$G^k[\alpha]$中无孤立点，
那么$u$有至少一个无故障邻居，令它为$y$。
根据引理~\ref{oneComponentConnectsone}
存在一条以$u$为起点的无故障路径$P$连通到子图$G^k[\beta]$。
假设$z$为路径$P$的终点，如果$z=v$，那么路径$P$即为$u$和$v$ 在图$G-F$中的一条无故障路径。
否则的话，因为$G^r[\beta]$是连通的，并且$z$和$v$都在子图$G^r[\beta]$中，
所以在$z$和$v$之间存在一条无故障路径$P'$，
那么路径$(P,P')$即为$u$和$v$在图$G-F$中的一条无故障路径(如图\ref{example1_2_1})。

情形1.2.2. $m_{r}>m_{r-1}$且$|F_{\alpha}\cup F_{\beta}| \geq 3\kappa(G)-2m_k-2m_r+6$
或$m_{r}=m_{r-1}$且$|F_{\alpha}\cup F_{\beta}| \geq 3\kappa(G)-2m_k-2m_r+5$。

若$m_{r}>m_{r-1}$，
我们有$|F\setminus (F_{\alpha}\cup F_{\beta})|\leq(3\kappa(G)-2m_r-1)-(3\kappa(G)-2m_r-2m_r+6)= 3m_k-7$；
若$m_{r}=m_{r-1}$，
我们有$|F\setminus (F_{\alpha}\cup F_{\beta})|\leq(3\kappa(G)-2m_r-2)-(3\kappa(G)-2m_r-2m_r+5)= 3m_k-7$。
那么因为$G\setminus (G^k[\alpha]\cup G^k[\beta])$中，
每个子图为$G^k[i]$，其中$i\in \langle m_k\rangle\setminus \{0,\alpha,\beta\}$,
共有$m_k-2$个子图。
那么，我们假设每个子图中的故障点数都大于等于$3$。
则有$|F\setminus (F_{\alpha}\cup F_{\beta})|\geq 3(m_k-2)=3m_k-6>3m_k-7$，
这和$|F\setminus (F_{\alpha}\cup F_{\beta})|\leq3m_k-7$相矛盾。
因此一定存在一个子图的故障顶点数小于等于$2$。我们不妨假设该子图为$G^k[\gamma]$。
其中$\gamma \in \langle m_{r}-1\rangle \setminus \{\alpha,\beta\}$。
因为子图$G^k[\beta]-F_{\beta}$中不存在孤立点和孤立边，
我们假设存在$u_1,u_2$使得$(u,u_1)\subset E(G)$并且$(u_1,u_2)\subset E(G)$。
同时我们假设$N_{G^k[\gamma]}(\{u,u_1,u_2\})=\{u',u_1',u_2'\}$。
因为$|F_{\gamma}|\leq 2$，
那么$\{u',u_1',u_2'\}$中至少有一个顶点是无故障的。
我们假设$u'$是无故障的,从$u'$到$u$之间的路径定义为$P_1$。
同理，对于顶点$v$,我们假设$v'$是无故障的,从$v'$到$v$之间的路径定义为$P_3$。
因为$|F_{\gamma}|<3m_k-7<\kappa^1(G^{k}[\gamma])$,
那么子图$G^{k}[\gamma]$是连通的，
则在$G^k[\gamma]-F_{\gamma}$中在$u'$和$v'$之间至少存在$1$条路径$P_2$。
因此，在图$G-F$中，路径$(P_1,P_2,P_3)$是$u$和$v$之间的一条无故障路径(如图\ref{example1_2_2})。

所以$G^r[\alpha]-F_\alpha$是连通的，得证。

另外我们考虑$|F_{\alpha}|\geq \kappa^1(G^{k}[\alpha])$的情况。
令$J=\langle m_r-1 \rangle \setminus \{\alpha\}$，
则$G^k[J]=G-G^k[\alpha]$。
根据引理~\ref{oneComponentConnectsToOthers}，
我们可以得到，顶点$u$可以通过一条无故障路径连通到$G^k[J]$图中的一个顶点，
我们假设该顶点为$u'$，假设该路径为$P_1$。
同理，顶点$v$可以通过一条无故障路径连通到$G^k[J]$图中的一个顶点，
我们假设该顶点为$v'$，假设该路径为$P_3$。
接下来，因为$|F_{\alpha}|\geq \kappa^1(G^{k}[\alpha])=2\kappa-2m_{k}-m_r+2$，
如果$m_r>m_{r-1}$，
那么我们有$|F_{J}|=|F-F_{\alpha}|\leq (3\kappa-2m_r-1)-(2\kappa-2m_{k}-m_r+2)
=\kappa-m_r+2m_{k}-3$。
又$\kappa^1(G^k[J])=2(\kappa-1)-m_r=2\kappa-m_r-2$，
我们可以得到$\kappa^1(G^k[J])-|F_{J}|=\kappa-2m_{k}+1$，
显然$\kappa-2m_{k}+1>0$，所以我们有$\kappa^1(G^k[J])>|F_{J}|$。
如果$m_r=m_{r-1}$，
同理我们有$|F_{J}|=|F-F_{\alpha}|\leq (3\kappa-2m_r-2)-(2\kappa-2m_{k}-m_r+2)
=\kappa-m_r+2m_{k}-4$。
又$\kappa^1(G^k[J])=2(\kappa-1)-m_r=2\kappa-m_r-2$，
我们可以得到$\kappa^1(G^k[J])-|F_{J}|=\kappa-2m_{k}$，
显然$\kappa-2m_{k}>0$，所以我们有$\kappa^1(G^k[J])>|F_{J}|$。
又因为$G^k[J]$中每个顶点都至少有一个无故障邻居，
我们可以得到$G^r[J]-F_{J}=G-V(G^r[\alpha])-F$是连通的。
则在图$G^r[J]-F_{J}$中$u'$和$v'$之间存在一条无故障路径$P_2$。
因此，在$G-F$中，路径$(P_1,P_2,P_3)$是$u$到$v$的一条无故障路径(如图\ref{example2})。

\begin{thm}
  \label{smallerthan11}
图$G(m_{r},m_{r-1},\ldots,m_{1})$的$2-$外连通度
当$m_{r}>m_{r-1}$，有$\kappa_{2}(G)\leq 3\kappa(G)-2m_{r}$；
当$m_{r}=m_{r-1}$，有$\kappa_{2}(G)\leq 3\kappa(G)-2m_{r}-1$，
其中$r \geq 5$，并且对于所有的$i \in \langle r \rangle \setminus \{0\}$，
都有$m_{i} \geq 3$。
\end{thm}

\begin{proof}

情形1. $m_{r}>m_{r-1}$。

令$u=u_{r}u_{r-1}\dots u_{1}$，
$v=v_{r}v_{r-1}\dots v_{1}$和
$w=w_{r}w_{r-1}\dots w_{1}$是图$G$中$3$个顶点，
满足$\{(u,v),(v,w),(v,w)\}\subset E(G)$
且对于每个$i \in \langle r \rangle \setminus \{0,r\}$,
有$u_r\neq v_r\neq w_r$,$u_{i}=v_{i}=w_{i}$。
则$|N_{G}(u,v,w)|=3(\kappa-(m_r-1))+(m_r-3)=3\kappa-2m_r$。
令$F=N_{G}(\{u,v,w\})$,
则我们可以把$G-F$分为两个部分,
其中一个是$\{u,v,w\}$,
另一个是$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$。
并且, 可以得到$G-F$中的每个分支都至少有$3$个顶点。
首先，$\{u,v,w\}$是一个包含三个顶点的连通子图。
所以，我们只需考虑$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$。

情形2. $m_{r}=m_{r-1}$。

令$u=u_{r}u_{r-1}\dots u_{1}$，
$v=v_{r}v_{r-1}\dots v_{1}$和
$w=w_{r}w_{r-1}\dots w_{1}$是图$G$中$3$个顶点，
满足$\{(u,v),(u,w)\}\subset E(G)$
且对于每个$i \in \langle r \rangle \setminus \{0,r\}$,
有$u_r\neq v_r$,$u_{i}=v_{i}$;
对于每个$i \in \langle r \rangle \setminus \{0,r-1\}$,
有$u_{r-1}\neq w_{r-1}$,$u_{i}=w_{i}$;
则$|N_{G}(u,v,w)|=3\kappa-2m_r-1$。

综合情形1和情形2，令$F=N_{G}(\{u,v,w\})$,
则我们可以把$G-F$分为两个部分,
其中一个是$\{u,v,w\}$,
另一个是$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$。
并且, 可以得到$G-F$中的每个分支都至少有$3$个顶点。
首先，$\{u,v,w\}$是一个包含三个顶点的连通子图。
所以，我们只需考虑$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$。
接下来我们考虑以下两种情况。

情形1. 假设$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$中包含孤立点。

根据假设，我们令该孤立点为$x$，那么我们有$N_{G}(\{x\})\subset F$，
即$N_{G}(\{x\})=N_{G}(\{x\})\cap F=N_{G}(\{x\})\cap N_{G}(\{u,v,w\})$。
根据引理\ref{cn3}我们有$|N_{G}(\{x\})|=|N_{G}(\{x\})\cap N_{G}(\{u,v,w\}|\leq4$，
又根据广义超立方体的定义，我们有$|N_{G}(\{x\})|=\kappa(G)\geq10$。
得出矛盾，所以假设不成立。

情形2. 假设$G-(N_{G}(\{u,v,w\}) \cup \{u,v,w\})$中包含孤立边。

根据假设，我们令该孤立边为$(x,y)$，那么我们有$N_{G}(\{x,y\})\subset F$，
即$N_{G}(\{x,y\})=N_{G}(\{x,y\})\cap F=N_{G}(\{x,y\})\cap N_{G}(\{u,v,w\})$。
根据引理\ref{cn4}，我们有$|N_{G}(\{x\})|=|N_{G}(\{x,y\})\cap N_{G}(\{u,v,w\}|\leq 8$，
又根据广义超立方体的定义，我们有$|N_{G}(\{x\})|=\kappa(G)\geq 17$。
得出矛盾，所以假设不成立。
综合情况1和情况2，我们可以得到，$G-(N_{G}(\{u,v,w\})$和$\{u,v,w\}$
两部分中无孤立点和孤立边，也就是每个分支中都包含至少$3$个顶点。

综上所述，当$|F|=|N_{G}(\{u,v,w\})|=3\kappa-2m_r$时，
图$G-F$是不连通的，且每个分支至少包含$3$个顶点。
所以，我们可以得到$\kappa_{2}(G)\leq 3\kappa(G)-2m_{r}$，得证。
\end{proof}

综合定理\ref{largerthan}和定理\ref{smallerthan11}，我们可以得到
除去$G(2,2)$和$G(2,3)$以外，
$r-$维广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$的
$1$-限制连通度：$\kappa^{1}(G(m_{r},m_{r-1},\ldots,m_{1}))=2\kappa-n$，
其中$r\geq2$。

\section{广义超立方体的$2$-外连通度下的容错路由}

我们在第三章给出的GHCFP算法可以在广义超立方体的总故障顶点数小于其$1$-限制连通度
且每个无故障顶点都至少有一个无故障邻居条件时，
构造$G(m_{r},m_{r-1},\ldots,m_{1})$中任意两个无故障顶点之间的一条无故障路径\ref{alg:ghcfp}。

在本小节中，我们给出当广义超立方体的总故障顶点数小于其$2$-外连通度时，
在每个剩余连通分支中至少包含$3$个无故障顶点条件下的容错路由算法GHCFP2。
之后，我们会分析并证明该算法的时间复杂度。

\subsection{$2$-外连通度条件下的容错路由算法}
%\algnewcommand\AND{\textbf{and~}}
%\algnewcommand\OR{\textbf{or~}}

\begin{breakablealgorithm}
\small
\label{alg:ghcfp2}
    \caption{GHCFP2($r,u,v,F,G(m_{r},m_{r-1},\ldots,m_{1})$)}
\begin{algorithmic}[1]
    \Require
    $r$-维广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$，
    一个点集合$F \subset V(G)$，
    两个不同的顶点$u=u_{r}u_{r-1}\dots u_{1}$且$v=v_{r}v_{r-1}\dots v_{1}$。
    \Ensure  图$G-F$中，一条以$u$起点为且$v$为终点的无故障路径。
    \Function{GHCFP2}{$r,u,v,F,G(m_{r},m_{r-1},\ldots,m_{1})$} %\GHCHP
    \State $G\gets G(m_{r},m_{-1},\ldots,m_{1})$;
    \If{$F= \emptyset $}  \ \Return GHCP$(r,u,v,G,F)$;
    \ElsIf{$(u,v)\in E(G)$}   \ \Return ($u,v$);
    \EndIf
    %\State $n\gets$ max $\{m_{1},m_{2},\ldots, m_{r}\}$;
    \State $k \gets$ GetDimension$(G)$;

    \State $F_{\alpha}\gets F \cap V(G^{k}[\alpha])$, $F_{\beta}\gets F \cap V(G^{k}[\beta])$;
    \If{$ \alpha \neq \beta $}
        \If {$|F_{\alpha}|\geq \kappa_{2}$}
            \State $P_1\gets$ PATHSEQ $(u, F, G^k[\alpha],G - V(G^{k}[\alpha]))$;
            \State 令$z$为路径$P_1$的最后一个顶点;
            \If{$z=v$} \ \Return $P_1$;
            \EndIf

        \ElsIf{$|F_{\beta}|\geq \kappa_{2}$}
            \State $P_1\gets$ PATHSEQ $(v, F, G^k[\beta],G - V(G^{k}[\beta]))$;
            \State 令$z$为路径$P_1$的最后一个顶点;
            \If{$z=v$} \ \Return $P_1$;
            \EndIf
            \State \Return (GHCFP$(r,u,z,G - V(G^{k}[\beta]), F\setminus F_{\beta})$, $P_1^{-1}$);
        \ElsIf{$m_{r}>m_{r-1}$ and $|F_{\alpha}|+|F_{\beta}|\leq 3\kappa_{1}-2m_{k}-2m_{r}+5$ or $m_{r}=m_{r-1}$ and $|F_{\alpha}|+|F_{\beta}|\leq 3\kappa_{1}-2m_{k}-2m_{r}+4$}
            \State $P_1\gets$ PATHSEQ $(u, F_{\alpha}\cup F_{\beta},G^k[\alpha],G^k[\beta])$
            \State 令$z$为路径$P_1$的最后一个顶点;
            \If{$z=v$} \ \Return $P_1$;
            \EndIf
            \State \Return($P_1$, GHCFP$(r,z,v,G^k[\beta],F_{\beta}$));
        \Else
            \State 选取一个子图$G^k[\gamma]$，使得$|F\cap G^k[\gamma]|=$ min $F\cap G^k[i]$，其中$\gamma, i \in \langle m_k-1 \rangle\setminus \{\alpha,\beta\}$;
            \State $F_{\gamma}\gets F \cap V(G^{k}[\gamma])$;
            \State $P_1\gets$ PATHSEQ $(u, F_{\alpha}\cup F_{\gamma}, G^k[\alpha], G^k[\gamma])$
            \State 令$x$为路径$P_1$的最后一个顶点;
            \State $P_2\gets$ PATHSEQ $(v, F_{\beta}\cup F_{\gamma}, G^k[\beta], G^k[\gamma])$;
            \State 令$y$为路径$P_2$的最后一个顶点;
            \If{$x=y$} \ \Return ($P_1$, $P_{2}^{-1}$);
            \EndIf
            \State \ \Return ($P_{1}$, GHCFP$(r,x,y,G_{1},F\cap V(G^k[\gamma]))$, $P_{2}^{-1})$;
        \EndIf
    \ElsIf{$|F_{\alpha}|< \kappa_{2}$} \ \Return (GHCFP$(r,u,v,G^k[\alpha],F_{\alpha})$);
    \Else
        \State $P_1\gets$ PATHSEQ $(u, F, G^k[\alpha], G - V(G^{k}[\alpha]))$;
        \State 令$x$为路径$P_1$的最后一个顶点;
        \State $P_2\gets$ PATHSEQ $(v, F, G^k[\alpha], G - V(G^{k}[\alpha]))$;
        \State 令$y$为路径$P_2$的最后一个顶点;
        \If{$z$为路径$P_{1}$和路径$P_{2}$的第一个公共顶点}
        \State \ \Return (Path$(P_{1},u,z)$, Path$(P_{2}^{-1},z,v))$;
        \EndIf
        \State \ \Return ($P_1$,GHCFP$(r, x, y, G - V(G^{r}[\alpha]), F \setminus F_{\alpha})$, $P_{2}^{-1}$);
   \EndIf
\EndFunction%

 \Function{PATHSEQ}{$u,F,H_{1},H_{2}$} %\P
    \If {$N_{G}(u)\cap V(H_{2}) \setminus F \neq \emptyset$}
        \State 从$N_{G}(u)\cap V(H_{2}) \setminus F$中选择一个顶点$u_{1}$;
        \State \Return $(u,u_{1})$;
    \Else
        \ForAll {$u_{1} \in N_{G}(u)\cap V(H_{1})\setminus F$}
            \If {$N_{G}(u_{1})\cap V(H_{2}) \setminus F \neq \emptyset$}
                \State 从$N_{G}(u_{1})\cap V(H_{2}) \setminus F$ 中选择一个顶点$u_{2}$;
                \State \Return $(u,u_{1}, u_{2})$;
            \EndIf
        \EndFor
    \EndIf
    \State 从$N_{G}(u)\cap V(H_{1}) \setminus F$中选择一个顶点$u_{1}$;
    \State $S \gets N_{G}(u_1)\cap V(H_{1})\setminus (F \cup (N_{G}(u)\cap V(H_{1}))$;
    \ForAll {$u_{2} \in S$}
        \If {$N_{G}(u_{2})\cap V(H_{2}) \setminus F = \emptyset$}
            \State 从$N_{G}(u_{2})\cap V(H_{2}) \setminus F$中选择一个顶点$u_{3}$;
            \State \Return $(u,u_{1},u_{2},u_{3})$;
        \EndIf
    \EndFor
    \State 从$N_{G}(u)\cap V(H_{1})\setminus F$中选择一个顶点$u_{1}$;
    \State $S \gets N_{G}(u_1)\cap V(H_{1})\setminus (F \cup (N_{G}(u)\cap V(H_{1}))$;
    \State 从$S$中选择一个顶点$u_{2}$;
    \State $S \gets N_{G}(u_2)\cap V(H_{1})\setminus (F \cup (N_{G}(\{u,u_1\})\cap V(H_{1})))$;
    \ForAll {$u_{3} \in S$}
        \If {$N_{G}(u_{3})\cap V(H_{2}) \setminus F = \emptyset$}
            \State 从$N_{G}(u_{3})\cap V(H_{2}) \setminus F$中选择一个顶点$u_{4}$;
            \State \Return $(u,u_{1},u_{2},u_{3},u_{4})$;
        \EndIf
    \EndFor
 \EndFunction%

  \Function{GetDimension}{$G$} %\P
    \State $S \gets \emptyset$;
    \ForAll {$u \in V(G)$ }
        \If{$|N_{G}(u)|= 1$}
            \State $S$集合中添加和u相关联的边所对应的维度;
        \EndIf
    \EndFor
    \State \ \Return min$(\{m_{1},m_{2},...m_{r}\} \setminus S $;
  \EndFunction%

\subsection{容错路由算法分析}
在分析该算法的时间复杂度之前，
我们先来分析函数$PATHSEQ(u,F,H_{1},H_{2})$和
函数$GHCP(x,y,G',F)$和$GetDimension(G)$的时间复杂度。
对于函数$GetDimension(G)$，
该函数遍历图$G$中的$N$个顶点，
每次遍历进行$\kappa(G)$次判断来得到该顶点的度，
如果该顶点度数为$1$,
则将度数为1的顶点相关联的邻边对应的维度添加到集合S中，
添加到集合的时间复杂度为$O(1)$。
则该循环的时间复杂度为$N \times \kappa(G)$。
之后取$k$的时间复杂度为$O(1)$。
所以该函数的时间复杂度为$N \times \kappa(G)+O(1)$=$N \times \kappa(G)$。

然后，对于函数$PATHSEQ(u,F,H_{1},H_{2})$，
其中$H_{1}$和$H_{2}$是图$G$的两个子图且$u$在图$H_{1}$中。
依据不同的情况，该函数可能会给出$u$到子图$H_{2}$的路径$(u,u_1)$、$(u,u_1,u_2)$、$(u,u_1,u_2,u_3)$
或者$(u,u_1,u_2,u_3)$。不同的情形下，函数执行的时间复杂度如下：
情形1. 得到的路径为$(u,u_1)$。
如果$N_{G}(u)\cap V(H_{2}) \setminus F \neq \emptyset$，
那么该函数需要从$N_{G}(u)\cap V(H_{2}) \setminus F$中选择一个顶点$u_{1}$，
然后该函数$PATHSEQ$返回的路径长度是$(u,u_1)$。
该情况的时间复杂度为$O(N_{G}(u)\cap V(H_{2}))< O(\kappa(G))$;
情形2. 得到的路径为$(u,u_1,u_2)$。
该函数会遍历集合$N_{G}(u)\cap V(H_{1})\setminus F$，取集合中每个顶点为$u_{1}$，
然后从$N_{G}(u_{1})\cap V(H_{2}) \setminus F$ 中选择一个顶点$u_{2}$，
然后该函数$PATHSEQ$返回的路径长度是$(u,u_1,u_2)$。
该情况的时间复杂度为$(N_{G}(u)\cap V(H_{1}))\times (N_{G}(u_{1})\cap V(H_{2}))< O(\kappa(G)^2)$;
情形3. 得到的路径为$(u,u_1,u_2,u_3)$。
在该情形下， 该函数首先会从$N_{G}(u)\cap V(H_{1}) \setminus F$ 中选择一个顶点$u_{1}$
并且得到集合$S \gets N_{G}(u_1)\cap V(H_{1})\setminus (F \cup (N_{G}(u)\cap V(H_{1}))$。
该步的时间复杂度为$0(N_{G}(u)\cap V(H_{1}) \setminus F)<O(\kappa(G))$。
然后该函数会遍历集合$S$，取集合中每个顶点为$u_{2}$，
然后从$N_{G}(u_{2})\cap V(H_{2}) \setminus F$中选择一个顶点$u_{3}$，
最后该函数$PATHSEQ$返回的路径$(u,u_1,u_2,u_3)$。
该情况的时间复杂度为$(O(\kappa(G)+O(\kappa(G)\times O(N_{G}(u_{2})\cap V(H_{2})))< O(\kappa(G)^2)$。
情形4. 得到的路径为$(u,u_1,u_2,u_3,u_4)$。
在该情形下， 该函数首先会从$N_{G}(u)\cap V(H_{1}) \setminus F$ 中选择一个顶点$u_{1}$
并且得到集合$S \gets N_{G}(u_1)\cap V(H_{1})\setminus (F \cup (N_{G}(u)\cap V(H_{1}))$。
然后该函数会从$S$ 中选择一个顶点$u_{2}$
并且得到集合$S \gets N_{G}(u_2)\cap V(H_{1})\setminus (F \cup (N_{G}(\{u,u_1\})\cap V(H_{1})))$。
该步的时间复杂度为$2 \times 0(N_{G}(u)\cap V(H_{1}) \setminus F)< 2 \times O(\kappa(G))$。
然后该函数会遍历集合$S$，取集合中每个顶点为$u_{3}$，
然后从$N_{G}(u_{3})\cap V(H_{2}) \setminus F$中选择一个顶点$u_{4}$，
最后该函数$PATHSEQ$返回的路径$(u,u_1,u_2,u_3,u_4)$。
该情况的时间复杂度为$(2\times O(\kappa(G)+O(\kappa(G)\times O(N_{G}(u_{3})\cap V(H_{2})))< O(\kappa(G)^2)$。
综合以上情况可得，函数$PATHSEQ(u,F,H_{1},H_{2})$的时间复杂度为$O(\kappa(G)^2$。

%$BuildPathSet(u,v)$ \cite{guo2009BCube}

%又有，该步执行的每一次都要判断路径中的每个顶点是否在故障集合$F$ 中。
%那么因为故障集合$|F|=O(2\times\kappa(G)-n-1)$，
%并且由于广义超立方体$G'$的故障直径为$O(\kappa(G')+1)$，

接下来，我们分析本算法~\ref{alg:ghcfp2}的时间复杂度。
在情形$\alpha \neq \beta$中，
如果$F_{\alpha}\geq \kappa_{2}$,
算法GHCFP会首先通过PATHSEQ$(u,G^k[\alpha],G - V(G^{r}[\alpha])$
得到一条从$u$到图$G-V(G^{r}[\alpha])$中顶点$z$的路径$P_{1}$，
该步的时间复杂度为$O(\kappa(G)^2)$。
然后，如果$z=v$，算法GHCFP将返回$P_{1}$，
那么得到图$G-F$中从$u$到$v$的一条无故障路径$P_{1}$的时间复杂度为
$O(\kappa(G)^3)$；
如果$z\neq v$，算法GHCFP会返回路径$(P_{1},GHCFP(r,z,v,G - V(G^{r}[\alpha],F\setminus F_{\beta})$。
然后运行函数$GHCFP(r,z,v,G - V(G^{r}[\alpha],F\setminus F_{\beta})$的时间复杂度为$O(\kappa(G)^3)$，
并且拼接路径$P_{1}$ 和路径$GHCFP(r,z,v,G - V(G^{r}[\alpha]),F\setminus F_{\beta})$的时间复杂度是$O(1)$。
因此，得到$G-F$图中从$u$到$v$的路径的时间复杂度为
$O(\kappa(G)^3+\kappa(G)^3+1)=O(\kappa(G)^3)$。
如果$|F_{\beta}|\geq \kappa_{2}$或者$|F_{\alpha}+F_{\beta}|\leq 2\kappa_{1}-2m_{r}-n+3$,
类似于在情形$|F_{\alpha}|\geq \kappa_{2}$中的讨论，
我们得到在$G-F$中一条从$u$到$v$无故障路径的时间复杂度为
$O(\kappa(G)^3)$。
如果 $m_{r}>m_{r-1}$ 和 $|F_{\alpha}|+|F_{\beta}|\leq 3\kappa_{1}-2m_{k}-2m_{r}+5$ 或者 $m_{r}=m_{r-1}$ 和 $|F_{\alpha}|+|F_{\beta}|\leq 3\kappa_{1}-2m_{k}-2m_{r}+4$，
算法GHCFP会首先选择一个满足条件$|F\cap G^k[\gamma]|=1$ 的子图$G^k[\gamma]$，其中
$i \in \langle m_r \rangle\setminus \{\alpha,\beta\}$。
令$G_{1}=G^k[\gamma]$，
由于$m_{k}-2<\kappa(G)$，
得到$G_{1}$的时间复杂度为 $O(m_{k}-2)<O(\kappa(G))$。
接下来，算法GHCFP将得到一条从$u$到子图$G_{1}$的顶点$x$的无故障路径$P_{1}$
和从$v$到子图$G_{1}$中的顶点$u$的一条无故障路径$P_{2}$，
得到这两条路径的时间按复杂度分别为$O(\kappa(G)^2)$。
然后，如果$x=y$，算法GHCFP将返回路径$P_{1}$。
因此，得到图$G-F$中一条从$u$到$v$的一条无故障路径的时间复杂度为
$ O(\kappa(G)+2\times\kappa(G)^2)=O(\kappa(G)^2)$；
如果$x\neq y$，算法GHCFP将返回路径$(P_{1},GHCFP(r,x,y,G_{1},F\cap V(G_{1}),P_{2}^{-1})$。
那么，得到图$G-F$中一条从$u$到$v$的一条无故障路径的时间复杂度为
$O(\kappa(G)+2\times \kappa(G)^2+\kappa(G)^3+1)=O(\kappa(G)^3)$。

在情形$\alpha = \beta$中，
如果$F_{\alpha}\leq \kappa_{2}$,
那么算法GHCFP将返回算法$GHCFP(u,v,G^k[\alpha],F_{\alpha})$，则时间复杂度为$O(\kappa(G)^3)$。
否则，算法GHCFP将得到一条以$u$为起点，以子图$G-V(G^k[\alpha])$中的$x$为终点的无故障路径$P_{1}$，
和从$v$到子图$G-V(G^k[\alpha])$中的顶点$y$的一条无故障路径$P_{2}$，
得到路径$P_{1}$和$P_{2}$的路径分别为$O(\kappa(G)^2)$。
然后，假设$z$为路径$P_{1}$和$P_{2}$的第一个公共顶点，
那么算法GHCFP将返回路径(Path$(P_{1},u,z)$, Path$(P_{2}^{-1},z,v))$。
既然，得到路径$(P_{1},u,z)$的时间复杂度为$O(1)$，
那么得到$G$图中从顶点$u$到$v$的时间复杂度为
$O(2\times\kappa(G)^2+O(1)=O(\kappa(G)^2)$；
若$P_{1}$和$P_{2}$无公共顶点，算法 GHCFP会返回路径
$(P_{1},GHCFP(r,x,y,G - V(G^{r}[\alpha]),F \setminus F_{\alpha}),P_{2}^{-1})$。
既然得到路径Path$(P_{1},u,z)$ 或者Path$(P_{2}^{-1},z,v)$)的时间复杂度为$O(1)$，
那么运行算法$GHCFP(r,x,y,G - V(G^{r}[\alpha]),F \setminus F_{\alpha})$
的时间复杂度为 $O(\kappa(G)^3)$。
因此, 得到$G$图中从$u$到$v$的一条无故障路径的时间复杂度为
$O(2\times\kappa(G)^2+\kappa(G)^3+2\times 1)=O(\kappa(G)^3)$。

综上所述， 本章的算法GHCFP2的时间复杂度为$O(\kappa(G)^3)$。

\section{本章小结}

本章基于外连通度的定义，
我们求出了广义超立方体的$1$-外连通度和$2$-外连通度，
并给出满足故障点数小于$2$-外连通度，且故障网络中每个分支至少包含3个顶点条件的
容错路由算法。
并取得如下研究成果：

1. 对于一个$r$-维广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$，
取$n=\max\{m_{1},m_{2},\ldots,m_{r}\}$，
则该广义超立方体的$1$-外连通度是$2\kappa(G)-n$。

2. 对于广义超立方体$G(m_{r},m_{r-1},\ldots,m_{1})$，
其中$r \geq 5$且对于每个$i \in \langle r \rangle \setminus \{0\}$ 都有$m_{i}\geq 3$
且$m_{i}\geq m_{i-1}$。
当$m_{r}>m_{r-1}$,
则广义超立方体的$2$-外连通度为$3\kappa(G) - 2m_{r}$；
当$m_{r}=m_{r-1}$,
则那么该广义超立方体的$2$-外连通度为$3\kappa(G) - 2m_{r}-1$。

3. 给出一个基于$2$-外连通度，时间复杂度为$O(\kappa(G)^{3})$的
无故障单播路由算法。
